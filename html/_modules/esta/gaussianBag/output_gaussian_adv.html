        print('writing data in xyz format')

[docs]    def get_forces(self):
        '''
        find the forces on all atoms 
        max and min of all forces on atoms
        
        parameters: 
                self.filename: str
                 name of the gaussian log file

        '''
    
        with open(self.filename, 'r') as f:
            lines = f.readlines()
            scfdone = False
            optdone = False
            
            nenergy = 0
            energies = []

            count_forces = 0
            
            #lthermochem = False
            #elec_and_zpe =[]
            
            #time_start = time.time()
            for i in range(0, len(lines)):
                
                if "SCF Done: " in lines[i]:
                    scfdone = True
                    nenergy += 1
                    energies.append(lines[i].split()[4])
                    
                if "Forces (Hartrees/Bohr)" in lines[i]: 
                    # reading all matches, but store only final forces
                    count_forces += 1
                    
                    
                if 'Optimization completed' in lines[i]:
                    #print('opt done !!')
                    optdone = True       
                
                if " NAtoms=" in lines[i]:
                    natoms = int(lines[i].split()[1])
                
            print ('+===+'*10)    
            print ('natoms: {}'.format(natoms))
            print('scf done: {}'.format(scfdone))
            print('no. of scf cycles: {}'.format(nenergy))
            print('no. of force cycles: {}'.format(count_forces))
            print('opt completed: {}'.format(optdone))
            print ('+===+'*10)    

            
            if optdone:
                forces = np.zeros((natoms, 3))
                count_forces_dummy = 0
                for i in range(len(lines)):
                    #count_forces_dummy = 0
                    if "Forces (Hartrees/Bohr)" in lines[i]: 
                        # reading all matches, but store only final forces
                        count_forces_dummy += 1
                        #print('no. of force cycles: {}'.format(count_forces))
                        
                        if count_forces_dummy == count_forces:
                            for j in range(natoms):
                                forces [j,0] = lines[i+3+j].split()[2]
                                forces [j,1] = lines[i+3+j].split()[3]
                                forces [j,2] = lines[i+3+j].split()[4]
                            
                    if "Cartesian Forces:" in lines[i]:
                        forces_max_rms = lines[i].split()
                        forces_max = float (forces_max_rms[3])
                        forces_rms = float (forces_max_rms[5])
                        

                print('count_forces_dummy: {}'.format(count_forces_dummy))
                print('count_forces: {}'.format(count_forces))
                
        if optdone:
            return forces_max, forces_rms, forces  # units forces Hartee/Bohr
        else:
            return None

                if "Coordinates (Angstroms)" in lines[i]:

                        icoord += 1
                    icoord_list.append(icoord)
            #print('Atoms list: {}'.format(icoord_list))
