def _bond_length(atomlist1, atomlist2):
    """
    returns element objects for the two atomic lists

    parameters
    ----------
    atomlist1: list 
        list of strings of atomic symbols
    atomlist2: list 
        list of strings of atomic symbols

    returns
    -------
    element_obj1: object
        element object for atomic list1
    element_obj2: object
        element object for atomic list2

    """

    import numpy as np
    import esta.element as element
    import sys
    
    if len(atomlist1) != len(atomlist2):
        print('# of atoms in the two list must be same .. exiting!!')
        sys.exit()
    
    
    element_obj1 = element.elements_data(atomlist1)
    element_obj2 = element.elements_data(atomlist2)
    
    return element_obj1, element_obj2

    bond_length: array

    '''

    val2 = np.zeros(len(atomlist2))

    dist = np.zeros(len(atomlist1))
    #dist2 = np.zeros(len(atomlist2))

    
    element_obj1, element_obj2 = _bond_length(atomlist1, atomlist2)

    if element_property is None:
        element_property = 'covalent_radius'
    
    if element_property == 'covalent_radius':    
        for i in range(len(element_obj1)):
            val1[i] = element_obj1[i].covalent_radius
            #val1[i] = element_obj1[i].atomic_radius

            #print('covalent rad: {}'.format(val1[i]))

        for i in range(len(element_obj2)):
            val2[i] = element_obj2[i].covalent_radius
            #val2[i] = element_obj2[i].atomic_radius
            #val2[i] = element_obj2[i].metallic_radius
            #val2[i] = element_obj2[i].vdw_radius
            
            #print('covalent rad: {}'.format(val2[i]))
        for i in range(len(element_obj1)):
            dist[i] = (val1[i] + val2[i] )
