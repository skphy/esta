print('**-----------------------------------------------**')    
bohr=double(scconst.physical_constants['Bohr radius'][0])
angstrom = double(scconst.physical_constants['Angstrom star'][0])
#bohr2ang= angstrom/bohr
# bohr2ang = np.array(angstrom/bohr, dtype=np.float32)
bohr2ang = np.array(bohr/angstrom, dtype=np.float128)

ryd = double(scconst.physical_constants['Rydberg constant times hc in eV'][0])

print(('ryd2ev:', ryd2ev))

hartree = scconst.physical_constants['hartree-electron volt relationship'][0]
har2ev = np.array(hartree, np.float128)
print(('har2ev:',har2ev))

har2ryd = np.array(har2ev/ryd2ev, dtype=np.float128)

ryd2har = np.array(1./har2ryd, dtype=np.float128)
print(('ryd2har:',ryd2har))

lambda2nu = scconst.physical_constants['speed of light in vacuum'][0]
