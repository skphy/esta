[docs]def get_metric_tensor(cell_matrix):
    
    """
    finding the metric tensor for cell matrix
    
    Parameters
    ----------
    cell matrix : 3D array 
        cell or lattice information in the cell matrix
    
    Returns
    -------
    metric-tensor : 3D array 
        matrix of metric tensor
    """
    
    import numpy as np
    cell_matrix = np.asarray(cell_matrix, dtype = np.float)
    return np.dot(cell_matrix, cell_matrix.T)
