        ''' delete atomic strs/points along an arrangement of pts in an array 
        of shape(n, nat, 3) by a step/stride (default=2). n is no. of atomic-sts/points 
        of shape(nat,3), nat is the no. of atoms in each set.
        '''
        
        if self.dim_atm_pos != 3:
            raise Exception('dim of atomic positions array is not 3 .. stopping!!')

        #self.dim_atm_pos # dimension of atomic position array 
        if len(self.atomic_positions) == len(self.atomic_labels):
            print("atomic positions are consistent with atomc labels .. proceeding!")
            
        else: 
            raise Exception("atomic positions are in consistent with atomc labels .. stopping!!")
            
            
        if step is None:
            step = 2
            # it means 2 atomic-strs in `n` atomic array in nat*3 dimension will be removed
                        
        nst = 0    
        for i in range(0,len(self.atomic_positions), step):
            nst = nst + 1
        print('no. of new strs/array pts are: {}'.format(nst))
    
        # atomic position new
        new_pos = np.zeros((nst, np.shape(self.atomic_positions)[1],3 ))
        for j,  i in enumerate(range(0,len(self.atomic_positions), step)):
            new_pos[j,:,:] = self.atomic_positions[i,:,:]
        #print('shape of new_pos: {}'.format(np.shape(new_pos)))
        
        # atomic symbols new
        new_symb = []
        for i in range(0,len(self.atomic_positions),step):
            new_symb.append(self.atomic_labels[i])

        print('shape of new_pos: {}'.format(np.shape(new_pos)))
        print('shape of new_symb: {}'.format(np.shape(new_symb)))

        return new_pos, new_symb
