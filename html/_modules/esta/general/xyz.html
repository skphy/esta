[docs]    def read_xyz(self, filename):
        """
        Parameters 
        ---------- 
        filename : string
            name of xyz file containing all xyz atomic 
            positions and atomic lables

        Returns
        -------
        pos : array 
            array of shape (no_of_strs, natoms, 3)
        label : list
            list of atomic labels containing all atomic symbols
        """


        with open(filename) as f:
            lines = f.readlines()
            for line in range(len(lines)):
                if line == 0:
                    natom = int (lines[line].split()[0])
                    print('no. of atoms: {}'.format(natom))
                
                
        with open(filename) as f:
            lines = f.readlines()
            nstrs = 1 # assuming atleast one xyz data present

            icount = 0
            for line in range(len(lines)):
                while line == (natom + 2)*icount: # or try module operator; line modulo natom+2 
                    icount += 1 
                    nstrs += 1
        n_structures = nstrs -1
        
        print('# of atoms in a str: {}'.format(natom))
        print('# of strs: {}'.format(n_structures))
             
        pos = np.zeros((n_structures, natom, 3))
        label = [] # for accomodating all atomic symbols of pos variable
            
        with open(filename) as f:
            lines = f.readlines()
            iucount = 0  # universal counter (starting from 1 ) on lines spanning over all loops...great counter ahh ..!!
            for i in range(n_structures):
                iucount = iucount + 2  # first two lines of xyz file

                label_dummy = []     
                
                for j in range(natom):
                    iucount = iucount + 1
                    pos[i, j, 0] = float (lines[iucount-1].split()[1])  # iuncount -1 ; as py counts from 0
                    pos[i, j, 1] = float (lines[iucount-1].split()[2])  # iuncount -1 ; as py counts from 0
                    pos[i, j, 2] = float (lines[iucount-1].split()[3])  # iuncount -1 ; as py counts from 0
                    label_dummy.append(lines[iucount-1].split()[0])     # iuncount -1 ; as py counts from 0

                label.append(label_dummy)
                
        ltest  = False
#         ltest = True
        if ltest:
            print(len(label), len(label[0]))
            print(np.shape(pos), len(pos))
            print(pos[0,0:5,:])
            print(pos[1,0:5,:])
        
        #elf.pos = pos
        #self.label = label
        return pos, label # atomic positions of all xyz sts, atomic labels of all strs
