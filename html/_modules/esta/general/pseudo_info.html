    '''
    get ATOMIC_SPECIES information required for generating 
    qe-input file
    
    input:
        list of elements; 
    output:
        output file containing:
        label of elements, mass, and UPF files
        
    '''

    import glob
    from mendeleev import element
#     atomic_species = ['B', 'N', 'O', 'C', 'H', 'Pt']

    filename='_'.join(atomic_species)+'_atomic_species.out'
    nln='\n'
    with open(filename, 'w') as f:
        f.write('ATOMIC_SPECIES'+nln)
        for i in atomic_species:
            filess = glob.glob(str(i)+'.pbe-*kjpaw*')
            print(i, element(str(i)).mass, filess)
            string = "  ".join(i for i in filess)
            f.write(str(i)+'\t'+str(element(str(i)).mass)+'\t'+string+nln)


            
if __name__=='__main__':
