[docs]    def read_xyz(self, xyzfile):
        """
        Parameters 
        ----------
        string : str
            name of xyz file, xyzfile
        
        Returns
        -------
        array : array of rank 2 with shape (natoms, 3)
            atomic positions
        list : list of strs 
            list of atomic labels, rank 1 array of atomic labels
            
        """    
        import numpy as np
        import sys
        
        self.xyzfile = xyzfile
        filename = self.xyzfile # extra assignment .. remove in future 
        
        print('filename {}:'.format(filename))
        print('getting no of atoms by reading xyz file:')
        with open(filename, 'r') as f:
            natoms = int(f.readline())
            print('no of atoms: {}'.format(natoms))

        lpos = False
        with open(filename, 'r') as f:
            natoms = int(f.readline())
            print('no of atoms: {}'.format(natoms))
            skipline = f.readline()
            lines = f.readlines()

            if len(lines) == natoms:
                lpos = True
                pos = np.zeros(natoms*3).reshape(natoms,3)
                print('continuing reading atoms locations ..')
                label = []
                for iat in range(natoms):
                    pos[iat:,0] = float((lines[iat].split())[1])
                    pos[iat:,1] = float((lines[iat].split())[2])
                    pos[iat:,2] = float((lines[iat].split())[3])
                    label.append(lines[iat].split()[0])
            else:   
                print('NOT all atomic positions present .. exiting!!')
                sys.exit()
                
        print('lpos: {}'.format(lpos))
        if lpos:
            self.natoms = natoms
            self.pos = pos
            self.label = label
            return (self.pos, self.label, self.natoms, "atm corrdinates in Angstrom")
        else:
            return None

    def _get_unique_list(self, inp):
            '''
            get unique elements from the input list `inp`

            '''
            ulist = []
            for x in inp:
                if x not in ulist:
                    ulist.append(x)
            return ulist

    def _get_grouped_list(self, llist):
        '''
        group same/identical string elements of the list 
        '''

        res = self._get_unique_list(llist)
        grouped_list = []
        idx = []

        for i in range(len(res)):
            for j in range(len(llist)):
                if res[i] == llist[j]:
                    idx.append(j)
        for i in idx:
            grouped_list.append(llist[i])

        return(grouped_list, idx)


    def _get_neach_type(self, inp):
        '''
        get a list of integers for each type of atoms # specifying their number
        '''    
        glist = self._get_grouped_list(inp)[0]
        ulist = self._get_unique_list(inp)
        #print(ulist)

        neach = []
        for i in ulist:
            k = 0
            for j in glist:
                if i == j:
                    k += 1
            neach.append(k)
        #print (neach)
        return(neach)

        '''
        get xyz data with frozan and not frozen atoms in the output file 

        parameters
        ----------
        frozen_tuple : tuple 
            tuple containg fixed atomic_positins, atomic_symbols, and atom index
        not_frozen_tuple : tuple 
            tuple containg not fixed atomic_positins, atomic_symbols, and atom index
        name_outfile: str, optional
            name of the output file
            
        returns
        -------
        name of xyz_frozen_file : str 
            create 'xyz_FROZEN' type file containing information of frozen and not frozen atoms etc

        '''


        from datetime import date
        today = date.today()

        #prefix = (self.infile.split('.'))[0]
        if name_outfile is None:
            outfile = 'xyz_Frozen'
        else:
            prefix = name_outfile
            outfile = prefix+'_Frozen'#'.xyz'

        pos_froz, symb_froz, atm_idx_froz = frozen_tuple
        pos_nfroz, symb_nfroz, atm_idx_nfroz = not_frozan_tuple
        
        
        
        print('writing data in xyz_Frozen format!!')   
        with open(outfile, 'w') as f:
            f.write(str(len(pos_froz) + len(pos_nfroz)))

            f.write('\nNot Frozen Atoms'+'\n')
            for j in range(len(pos_nfroz)):    
               f.write( str(symb_nfroz[j]).ljust(3) +"   "+str( " ".join(["{0:16.9f}".format(i) for i in pos_nfroz[j]]))+' '+str(atm_idx_nfroz[j])+'\n')        
            
            f.write('Frozen Atoms'+'\n')
            for j in range(len(pos_froz)):    
                f.write( str(symb_froz[j]).ljust(3) +"   "+str( " ".join(["{0:16.9f}".format(i) for i in pos_froz[j]]))+' '+str(atm_idx_froz[j])+'\n')
