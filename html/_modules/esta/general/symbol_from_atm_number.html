[docs]def get_symbol_from_atomic_number(self, atomic_num):
    '''
    return list of atomic symbols from list of atomic numbers
    '''

    from mendeleev import element
    #ele_obj = element([1,2])

    atomic_num_int = [ int(i) for i in atomic_num]

    ele_obj = element(atomic_num_int)
    print('atomic number list is: {}'.format(atomic_num_int))

    atomic_label = []
    for i in ele_obj:
        #print(i.symbol)
        atomic_label.append(i.symbol)
    print('atomic labels are : {}'.format(atomic_label))
    return atomic_label
