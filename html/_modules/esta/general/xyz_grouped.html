        '''
        find atom positions with labels
        ...

        '''
        infile = self.infile

        import numpy as np

    #     natoms = self.get_natoms(infile)[0]
        natoms = self.get_natoms()[0]

        with open(infile,'r') as f:

            lines = f.readlines()
            pos = np.zeros(natoms*3).reshape(natoms,3)
            lpos = False

            #print('last occurance of ATOMIC_POSITIONS/Forces')
            #print('     in pwo file is taken for position variable ...this is wrong')
            for i in range(0, len(lines)):
                if 'ATOMIC_POSITIONS (angstrom)' in lines[i]:
                    lpos = True
                    for j in range(natoms):
                        pos[j:,0] = float(lines[i+1+j].split()[1])
                        pos[j:,1] = float(lines[i+1+j].split()[2])
                        pos[j:,2] = float(lines[i+1+j].split()[3])

            pos_indx= []
            label =[]
            for j in range(len(lines)):
                if 'ATOMIC_POSITIONS (angstrom)' in lines[j]:
                    pos_indx.append(j)
            last_indx = pos_indx[-1]

            pos_ = np.zeros(natoms*3).reshape(natoms,3)
            for j in range(last_indx+1, last_indx+1+natoms):
                label.append((lines[j].split()[0]))
                k = j- (last_indx+1)
                pos_[k:,0] = float(lines[j].split()[1])
                pos_[k:,1] = float(lines[j].split()[2])
                pos_[k:,2] = float(lines[j].split()[3])
            #print('**'*10,k+1, '=no of atoms')

            if lpos:
                #print(' ')
                #print(len(label))
                #print (label)

                # Note pos variable is replaces by atomic positions many time
                #  depending upon the search critea (ATOMIC_POSITIONS); pos_ is updated
                # only once ..due to the new implementation...good sk!!
                # NOTE: try this way to other search conditions ..good luck

                #return (pos, pos_, label, "Angstrom")
                return (pos_, label, "Angstrom")

            else:
                print('\n------------------------------------')
                print("Serious error in file ..............")
                print("ATOMIC_POSITIONS block not present")
                print('------------------------------------\n')
                return None

[docs]    def get_grouped_list(self, llist):
        '''
        get same string elements of list grouped together
        '''

        def get_unique_list(inp):
            '''
            get unique elements of list

            return:
                    string for each type of atoms 
            '''
            ulist = []
            for x in inp:
                if x not in ulist:
                    ulist.append(x)
            return ulist

        res = self.get_unique_list(llist)
        grouped_list = []
        idx = []

        for i in range(len(res)):
            for j in range(len(llist)):
                if res[i] == llist[j]:
                    idx.append(j)
        for i in idx:
            grouped_list.append(llist[i])

        return(grouped_list, idx)

        '''
        get number of list entries of each type

        return:

        '''    
        glist = self.get_grouped_list(inp)[0]

        #print(ulist)

        neach = []
        for i in ulist:
            k = 0
            for j in glist:
                if i == j:
                    k += 1
            neach.append(k)
        #print (neach)
        return(neach)
