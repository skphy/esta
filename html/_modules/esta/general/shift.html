    '''
    method/function to shift atoms:
    - shift atoms in xyz file to origin 
    '''

    import numpy as np
    #     ffile = np.loadtxt('wire100.xyz', comments='reading xyz file', skiprows=2, usecols=range(1,4))
    ffile = np.loadtxt(filename,
                       comments='reading xyz file',
                       skiprows=2,
                       usecols=range(1, 4))
                       #usecols=range(0, 4)) error due to symbol

    natoms = np.shape(ffile)[0]
    #print 'no atoms:', natoms

    xcoordinates = (np.zeros(natoms * 3)).reshape(natoms, 3)
    for i in range(natoms):
        xcoordinates[i, :] = ffile[i, :] + np.array(
            [-np.min(ffile[:, 0]), -np.min(ffile[:, 1]), -np.min(ffile[:, 2])])

    with open(filename, 'r') as f:
        lines = f.readlines()
        atomlist = []
        #print('natoms: {}'.format(natoms))

    '''
    method/function to shift atoms:
    - shift atoms in xyz file along a direction
    '''

    import numpy as np
    import sys
    #ffile = np.loadtxt(filename, comments='reading xyz file', skiprows=2) #, usecols=range(1,4))
    #ffile = np.genfromtxt(filename, skip_header=2)

    with open(filename, 'r') as f:
        line = f.readlines()
        #print('lines in file', len(line))
        #print('atoms are 2 less than no of lines')
        natoms = len(line) - 2
        atomic_positions = np.zeros(3 * (len(line) - 2)).reshape(
            len(line) - 2, 3)
        labels = []
        for i in range(len(line)):
            if i > 1:
                labels.append((line[i]).split()[0])
                atomic_positions[i - 2, 0] = float((line[i]).split()[1])
                atomic_positions[i - 2, 1] = float((line[i]).split()[2])
                atomic_positions[i - 2, 2] = float((line[i]).split()[3])
        #print (labels)
        #atomic_positions[:,0] = atomic_positions[:,0] + xshift
        #atomic_positions[:,1] = atomic_positions[:,1] + yshift
        #atomic_positions[:,2] = atomic_positions[:,2] + zshift
        #print(atomic_positions)
        #print type(atomic_positions)
        atomic_positions = atomic_positions + np.array(
            [xshift, yshift, zshift])

    # uncomment the follwoing to get only the shifted coordinates
    #np.savetxt('xnew_'+filename, np.transpose([atomic_positions[:,0],\
    #                                          atomic_positions[:,1],\
    #                                          atomic_positions[:,2] ]), fmt='%10.6f')
    prefix = 'gsft'
    with open(prefix + filename, "w") as ff:
        #with open(filename+"_", "w") as ff:
        ff.write(str(natoms) + "\n")
        ff.write(str("\n"))
        for i in range(natoms):
            atmx = atomic_positions[i, 0]
            atmy = atomic_positions[i, 1]
            atmz = atomic_positions[i, 2]
            ff.write(labels[i] + "    " + str(atmx) + "    " + str(atmy) +
                     "    " + str(atmz) + "\n")
        print('saving newfile with atoms shifted along a direction:',
