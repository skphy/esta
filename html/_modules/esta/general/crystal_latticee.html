[docs]    def reciprocal_lattice(self):
        '''
        method to calculate the reciprocal lattice vectors        

        parameters
        ----------
        real space lattice vectors: LV1, LV2, LV3
        
        returns
        -------
        reciprocal space lattice vectors : b1, b2, b3 
        
        Notes
        -----
        a factor of 2pi is included in b1, b2, and b3
        '''
        import numpy as np

        LV1 = self.LV1
        LV2 = self.LV2
        LV3 = self.LV3
        real_volume = np.dot(LV1, np.cross(LV2,LV3))
        b1 = 2*np.pi * np.cross(LV2, LV3)/ real_volume
        b2 = 2*np.pi * np.cross(LV3, LV1)/ real_volume
        b3 = 2*np.pi * np.cross(LV1, LV2)/ real_volume
        
        #no need to add them here
        #self.b1 = b1
        #self.b2 = b2
        #self.b3 = b3

        return b1, b2, b3
