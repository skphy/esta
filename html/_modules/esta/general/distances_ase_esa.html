    dist_vectors = dist_ase[0]
    dist_len = dist_ase[1]

    if lprint_dist:
        for i, ipos in enumerate (range(len(pos1))):

                print('{0:3d} {1:3d} {2:5.2f}'.format(i+1, j+1, dist_len[ipos, jpos]))
    return dist_vectors, dist_len
