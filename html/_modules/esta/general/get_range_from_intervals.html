[docs]def get_range_int_from_intervals(list_of_intervals):
    """get range of integers from intervals 
    
    Parameters
    ----------
    list_of_intervals :  list of strings
        list of string of intervals, e.g. ['1','2-4','6-10']
        
    Returns
    -------
    elem : range of integers 
        range of integers specifying the intervals, e.g. [1, 2, 3, 4, 6, 7, 8, 9, 10]

    """

    #listt = ['1','2-4','6-10']
    ele = []
    #for i in listt:
    for i in list_of_intervals:
        if len(str(i).split("-")) == 2:
            val = str(i).split("-")
            val_ini = int(val[0])
            val_fin = int(val[1])
            #print(val_ini, val_fin)
            for j in range(val_ini , val_fin+1 ):
               ele.append(j) 
        else: 
            ele.append(i)
    elem = [int(e) for e in ele]
    return elem
