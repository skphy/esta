[docs]def reciprocal_lattice(LV1, LV2,LV3):
    '''
    input: real space lattice vectors: LV1, LV2, LV3
    output: reciprocal space lattice vectors: b1, b2, b3 
            - note: a factor of 2pi is included in b1, b2, and b3
    
    '''
    import numpy as np
    
    real_volume = np.dot(LV1, np.cross(LV2,LV3))
  
    #b1 = 2*np.pi * np.cross(LV2, LV3)/ np.dot(LV1, np.cross(LV2,LV3))
    #b2 = 2*np.pi * np.cross(LV3, LV1)/ np.dot(LV1, np.cross(LV2,LV3))
    #b3 = 2*np.pi * np.cross(LV1, LV2)/ np.dot(LV1, np.cross(LV2,LV3))

    b1 = 2*np.pi * np.cross(LV2, LV3)/ real_volume
    b2 = 2*np.pi * np.cross(LV3, LV1)/ real_volume
    b3 = 2*np.pi * np.cross(LV1, LV2)/ real_volume
    
    return b1, b2, b3
