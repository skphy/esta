#!/usr/bin/env python
# coding: utf-8

'''
read band.yaml file output from the phonopy code and write the specific 
phonon q point freqency in cm^{-1}.
'''


[docs]def read_yaml_band(filename = None, qpt_index = None, outfile = None):
    '''
    parameters
    ----------
    filename : str
        name of the filename containing band data in yaml format, default is band.yaml
    qpt_index : integer
        qpt index for which frequency data to be read, default is first qpoint.
    outfie : str
        outfile is name of the output file, default is band_cmii.dat
        
    Returns
    -------
    freq : array of reals
        vibration frequencies in cm-1 
    '''
    

    import yaml
    import sys
    import numpy as np

    import esta.measure.conversion_factors as cf

    

    #yaml_file = open("band.yaml")
    if filename is None: 
        filename = 'band.yaml'
    yaml_file = open(filename)
    
    if qpt_index is None: 
        qpt_index = 1
    
    print("*"*50)
    print('reading filename: {}'.format(filename))
    print('qpt index is {}'.format(qpt_index))
    print("*"*50)
    
    parsed_yaml_file = yaml.load(yaml_file, Loader=yaml.FullLoader)


    #print(parsed_yaml_file["phonon"])
    #dictt = parsed_yaml_file["phonon"][102]  ##102 or 103 number is gamma point, numbering from 1
    dictt = parsed_yaml_file["phonon"][qpt_index-1]  ##102 or 103 number is gamma point, numbering from 1
    #print(dictt)


    for k, v in dictt.items():
        #print(k, '-->',  v)
        if k == 'band':

            #print('freq in Thz from band.yaml file')        
            nfreq = len(v)
            print('no. of freqs are: {}'.format(nfreq))
            #sys.exit()


            freq = np.zeros(nfreq)
            for ii, i in enumerate(v):
                freq[ii] = float (i['frequency'])
                #print("{} freq(THz):{}".format(ii+1, freq[ii]))

    # converting freq in THz to cm-1            
    freq *= 1/cf.cmii2thtz       
    
    #filename = 'freq_gamma_cmii.dat'
    if outfile is None:
        outfile = 'band_cmii.dat'
    with open(outfile, 'w') as f:
        for i in range(len(freq)):
             f.write(  str.ljust(str(i+1),4) +"   "+  str("  ".join(["{0:8.3f}".format(freq[i])]))+'\n' )
    print ('writing freqs to file: {}'.format(outfile))    


if __name__== "__main__":
    read_yaml_band(qpt_index=102)
