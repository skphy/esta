[docs]def get_lup_inp(filename, lenergy=None):    
    '''
    read XXX_LUPINP.log file for getting the Path (xyz strs)
    
    Parameters 
    ----------
    filename : string
        filename, name of xxx_LUPINP.log file
        
    lenergy : logical, optional
        lenergy, whether energy to be extracted from the input filename
        
    
    Returns
    -------
    create xyz file containing all xyz strs with energy information in the comment if lenergy is true ..
    
    '''
    if lenergy is None:
        lenergy = False
        print('lenergy => energy information absent!!')
    print('lenergy True => energy information MUST be present in log file!!')
    
    with open(filename) as f:
        lines_list = f.readlines()
        print('no. of lines: {}'.format(len(lines_list)))
        
        
        n_node = 0
        for iline, line in enumerate(lines_list):
            if "NODE 0" in lines_list[iline]:
                #print(lines_list[iline])
                icount= 0
                atm_symbol = []
                while len(lines_list[iline+icount+1].split()) == 4:
                    atm_symbol.append(lines_list[iline+icount+1].split()[0])
                    icount +=1

                natoms = icount
                print('no. of atoms: {}'.format(icount))
                #print('atomic symbols: {}'.format(atm_symbol))
                print('no. of atomic symbols: {}'.format(len(atm_symbol)))
                if int(natoms) != len(atm_symbol):
                    print('Something wrong with atomic symbols and or no. of atoms!!')
                    sys.exit()
            
            if "# NODE" in lines_list[iline]:
                n_node += 1
        print('no. of nodes: {}'.format(n_node))
            
        
        position_all = np.zeros((n_node, natoms,3))
        #energies = np.zeros(n_node)
        n_node_dummy  = 0
        for iline, line in enumerate(lines_list): 
            
            if "# NODE" in lines_list[iline]:
                n_node_dummy += 1 
                for i in range(1, natoms+1):
                    position_all[n_node_dummy-1, i-1, 0] = float (lines_list[iline+i].split()[1])
                    position_all[n_node_dummy-1, i-1, 1] = float (lines_list[iline+i].split()[2])
                    position_all[n_node_dummy-1, i-1, 2] = float (lines_list[iline+i].split()[3])
                #energies[n_node_dummy-1] = float (lines_list[iline+natoms+2].split()[1])

        if lenergy:
            #energy = np.zeros(n_node) # as as no. of nodes found above
            energy  = []
            for iline, line in enumerate(lines_list): 
                if "ENERGY" in lines_list[iline]:
                    #print('ENERGY keyword present to extract the energy!!')
                    dummy = (lines_list[iline].split()[-3])
                    energy.append(float(dummy.split('(')[-1]))
            #print('energy: {}'.format(energy))
                
        ltest = False
        if ltest: 
            print('atomic positions all: {}'.format(position_all[-1,0:10,0]))
            print('shape of atomic positions all : {}'.format(np.shape(position_all)))
            #print('energies (Hartree), len(energies): {}, {}'.format(energies, len(energies)))
    
    if lenergy:
        return position_all, atm_symbol, energy # positions at nodes along with atomic symbols and energy
    else:
        return position_all, atm_symbol
