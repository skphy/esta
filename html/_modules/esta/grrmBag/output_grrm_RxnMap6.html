        '''
        get xyz files: atomic positions and atomic labels from "*_TS_list.log" file of GRRM17
        '''
        
        #filename = self.filename
        filename = logfile_name
        with open(filename, 'r') as f:
            lines=f.readlines()
            
            nstructures = 0
            for i in range(0, len(lines)):
                if "# Geometry of TS" in lines[i]:
                    nstructures += 1
            print('Number of TS (approx TS) structures found: {0}'.format(nstructures))

            lpos = False
            #natoms=0
            for i in range(len(lines)):
                #print('natoms:1--', natoms)
                if "# Geometry of TS" in lines[i]:
                    lpos = True
                    
                    natoms=0
                    #print('natoms:2--', natoms)
                    while len((lines[i+natoms+1]).split()) == 4 and len((lines[i+natoms+1]).split()[0]) != 12:
                        natoms += 1
                    #print('natoms:3--', natoms)
            print('no of atoms are: {}'.format(natoms))
            
            istr = 0
            pos_all = np.zeros(natoms*3*nstructures).reshape(nstructures,natoms,3) 
            for i in range(len(lines)):
                #print('istr: {}'.format(istr))
                if "# Geometry of TS" in lines[i]:
                    pos = np.zeros(natoms*3).reshape(natoms,3)
                    iat=0
                    label = []
                    #while len((lines[i+iat+1]).split()) == 4:
                    while len((lines[i+iat+1]).split()) == 4 and len((lines[i+iat+1]).split()[0]) != 12 :
                        pos[iat:,0] = float((lines[i+iat+1].split())[1])
                        pos[iat:,1] = float((lines[i+iat+1].split())[2])
                        pos[iat:,2] = float((lines[i+iat+1].split())[3])
                        #
                        #EXTREMELY IMPORTANT: label list is destroyed everytime
                        #                     a match in IF statement
                        #                     is found; only last label list is 
                        #                     left and therefore used 
                        #                     for labeling the atoms
                        label.append(lines[i+iat+1].split()[0])
                        #
                        pos_all[istr, iat:, 0] = pos[iat:, 0]
                        pos_all[istr, iat:, 1] = pos[iat:, 1]
                        pos_all[istr, iat:, 2] = pos[iat:, 2]
                        iat  += 1
                    istr += 1
                    #print('testing: MUST be equal')
                    #print('natoms =', natoms,'=','iat=',iat)
            #print('nstructures =', nstructures,'=','istr=',istr)

            if lpos:
                return (pos_all, label, "atm corrdinates in Angstrom")
            else:
                import sys
                print('\n------------------------------------')
                print("Serious error reading in *_TS_list.log GRRM file ...")
                print('------------------------------------\n')
                sys.exit()
                return None

        '''
        get Energy from *_TS_list.log file of all TS structures
        '''
        with open(self.filename, 'r') as f:
            lines=f.readlines()
            
            energy = []
            for i in range(0, len(lines)):
                if "Energy    =" in lines[i]:
                    energy.append(float(lines[i].split()[2]))
                #for new version of grrm
                if "Relat.Energy               =" in lines[i]:
                    energy.append(float(lines[i].split()[2]))
        return energy

    

[docs]    def get_energy_general(self, com_file_name, which_state='EQ'):
        '''
        get Energy of all TS/PT/EQ states from comfile_TS/PT/EQ _list.log file
        
        input:
        ------
            str: com file name with .com extension
            str: choose which states energies to get; EQ/TS/PT
        
        output:
        -------
            list: energies list of EQ/TS/PT structures
        
        '''
        import sys
        
        if com_file_name.split(".")[-1] != 'com':
            print('file name with .com extension is required!!')
            sys.exit()
        if which_state not in {'EQ', 'TS', 'PT'}:
            print('structures must be either from EQ or TS or PT xxx_EQ/TS/PT_list.log file!!')
            sys.exit()
            
        filename = com_file_name.split('.')[0] +"_"+str(which_state)+"_list.log"
        with open(filename, 'r') as f:
            lines=f.readlines()
            
            energy = []
            for i in range(0, len(lines)):
                if "Energy    =" in lines[i]:
                    energy.append(float(lines[i].split()[2]))
                #for new version of grrm
                if "Relat.Energy               =" in lines[i]:
                    energy.append(float(lines[i].split()[2]))
        return energy

    
    

[docs]    def get_connection(self, logfile_name):
        '''
        find which TS stats is connected to which EQ states
        
        input:
        ------
           str: filename 
        
        output:
        -------
            dictionary of TS and connections: {'TS': TS_list, 'Connection': connection}
            
            
        e.g:
        
        search the following is *TS_list.log file:
        
        # Geometry of TS 0, SYMMETRY = C1
        CONNECTION : 4 - 5
        '''
        print("\n")
        print('note--'*10)
        print('log file name Must be either xxx_TS_list.log or xxx_PT_list.log')
        print('Name of logfile: {}'.format(logfile_name))
        print('\n')
        #with open(self.filename, 'r') as f:
        with open(logfile_name, 'r') as f:            
            lines=f.readlines()
            
            TS_list = []
            connection = []
            for i in range(0, len(lines)):
                if "Geometry of TS" in lines[i]:
                    TS_list.append(lines[i].split(",")[0].split()[-1])
                if "CONNECTION " in lines[i]:
                    connection.append(lines[i].split()[2::2]) 
                
            ltest=False
            if ltest:
                print('TS_list: {}'.format(TS_list))
                print('connection list: {}'.format(connection))
            
            
        connection_dict = {'TS': TS_list, 'Connection': connection}
        return connection_dict
