[docs]    def read_xyz(self):
        """
        
        input: 
        ------
            filename, str, filename xyz file name
        
        output:
        ------
            atomic positions: array: rank 2 array of shape (natoms, 3)
            atomic labels: list/array of strings: rank 1 array of atomic labels
            no of atoms: scalar
        """    
        import numpy as np
        import sys
        
        filename = self.xyzfile
        with open(filename, 'r') as f:
            natoms = int(f.readline())
            print('no of atoms: {}'.format(natoms))
            skipline = f.readline()
            
            lpos = False
            lines = f.readlines()
            
            if len(lines) == natoms:
                lpos = True
                pos = np.zeros(natoms*3).reshape(natoms,3)
                print('continuing reading atoms locations ..')
                label = []
                for iat in range(natoms):
                    pos[iat:,0] = float((lines[iat].split())[1])
                    pos[iat:,1] = float((lines[iat].split())[2])
                    pos[iat:,2] = float((lines[iat].split())[3])
                    label.append(lines[iat].split()[0])
            else:   
                print('NOT all atomic positions present .. exiting!!')
                sys.exit()

            if lpos:
                self.pos = pos
                self.label = label
                return (self.pos, self.label, "atm corrdinates in Angstrom")
            else:
                return None

        e.g:
        
        
        Options
        sublink=/home/kumarsonu/bin/link_excode_allv3.1_.py

        DownDC=12

        '''

        prefix = self.xyzfile.split('.')[:-1][0]
        gfile =  str(prefix)+'_grrm'+str(self.cal_type)+'_flags.dat'
        
        #--------------------------------------------
        #print("*"*10)
        lprint=False
        #lprint=True
        
        with open(gfile,'w') as f:
            for key, value in kwargs.items(): 
                arg_match = key.split('_')[-1]
#               print(arg_match)
                if arg_match == 'arg':
                    if lprint:
                        print('arg:'.format(arg_match))
                        print('key: {}, value: {}'.format(key, value))
                    f.write(str(value)+'\n')
                if arg_match != 'arg':
                    if lprint:
                        print('val arg:'.format(arg_match))
                        print('key: {}, value: {}'.format(key, value))
                    f.write(str(key)+'='+str(value)+'\n')
        #test print
        #print('--c-c-c')
        #print(kwargs.items())
        print("*"*10)
        #--------------------------------------------
        
        return None

        '''
        input:
        ------
            atomic postions
            atomic labels
            no of atoms
        
        output:
        ------
            com file containing info of atomic structure
        
        '''
        import numpy as np
        from datetime import date
        today = date.today()

        pos = self.read_xyz()[0]
        label = self.read_xyz()[1]
        #pos, label = self.get_atomic_positions()[0], self.get_atomic_positions()[1]
    
        prefix = self.xyzfile.split('.')[:-1][0]
        # print(prefix)
        genfile =  str(prefix)+'_grrm'+str(self.cal_type)+'.com'
        # with open('')
        # print(genfile)
        with open(genfile, 'w') as f:
            if interface == 'orca':           
                f.write('%'+str('link=Orca')+'\n')
            else:
                f.write('%'+str('link=non-supported')+'\n')
                
            f.write('# '+str(self.cal_type)+'\n')
            f.write(' '+'\n')
            f.write(str(self.charge)+' '+str(self.multiplicity))
            for j in range(len(pos)):
                #f.write("\n"+str(label[j])+"\t"+str("\t".join(map(str, pos[j]))))
                f.write("\n"+str.ljust(label[j],3)+"  "+str("  ".join(["{0:16.9f}".format(i) for i in pos[j]])))
            f.write("\n")
            for key, value in kwargs.items(): 
                arg_match = key.split('_')[-1]
                if arg_match == 'arg':
                    f.write(str(value)+'\n')
                if arg_match != 'arg':
                    f.write(str(key)+'='+str(value)+'\n')
                
        print('writing com file for '+str(self.cal_type)+' type grrm calculation !!!')

[docs]    def write_inp(self, xyzfile, charge, multiplicity): #, interface = None, **kwargs):
        '''
        
        generate input file for ORCA like below:
        
        
        ! XTB2 NoUseSym @job

        * xyz  0 1
        @geometry
        *
        
        
        
        input:
        ------
            prefix: name of inp file with .inp extension
            charge: chare on ths system
            multiplicity: spin multiplicity (2S+1)

        output:
        ------
            inp file for orca containg charge and multiplicity options
            
        '''
        import numpy as np
        prefix = self.xyzfile.split('.')[:-1][0]
        genfile =  str(prefix)+'_grrm'+str(self.cal_type)+'.inp'
        
        str1 = '! XTB2 NoUseSym @job'
        str2 = ' '
        str3 = '* xyz  0 1'
        str4 = '@geometry'
        str5 = '*'
        with open(genfile, 'w') as f:
            f.write(str(str1)+'\n')
            f.write(str(str2)+'\n')
            f.write(str(str3)+'\n')
            f.write(str(str4)+'\n')
            f.write(str(str5)+'\n')
        print('writing inp file for '+str(self.cal_type)+' type grrm calc with ORCA interface!!!')
