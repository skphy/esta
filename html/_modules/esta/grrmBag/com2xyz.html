[docs]def get_coord(filename):
    '''
    get coord and atomic labesls from com file of gaussian or GRRM17 
    
    input: 
    ------ 
        input filename 
    
    output:
    ------
        scalar; no of atoms
        list;  for atomic labels
        array; for atomic positions
    '''
    import numpy as np
    
    with open(filename, 'r') as f:
        lines = f.readlines()
        
        icount = 0
        for line in lines:
            if len(line.split()) == 4: # and not any(ln for ln in line) == "=":
                #print(line.split(), len(line.split()))
                icount = icount + 1
        natom = icount
        
    pos = np.zeros((natom, 3))
    with open(filename, 'r') as f:
        lines = f.readlines()
        
        icount = 0
        label = []
        for iline, line in enumerate(lines):
            if len(line.split()) == 4: #  and not any(ln for ln in line) == "=":
                pos[icount,0] = float(line.split() [1])
                pos[icount,1] = float(line.split() [2])
                pos[icount,2] = float(line.split() [3])
                label.append(str(line.split() [0]))
                icount = icount + 1
            #print('iline {}'.format(iline))
            
    iverbosity = False
#     iverbosity = True
    if iverbosity:
        print('no of atoms are {}'.format(natom))
        print('atoimic labesl are {}'.format(label))
        print('atomic postions:\n {}'.format(pos))
        
    return natom, label, pos

[docs]def get_xyz_file(filename, filetype=None):
    '''
    get xyz file from com file
    
    input: 
    ------
          label: string list or string array
          pos: array of shape (natoms, 3)
          infile: input file name
          
    otuput:
    ------
         xyz file in the present dir; name of file derived from input file
    '''
    import numpy as np
    from datetime import date
    today = date.today()
    
    if filetype == 'kin':
        print('reading com file with kinetic options!!')
        natom, label, pos = get_coord_grrm_kin(filename)
    else:
        print('reading normal com file!!')
        natom, label, pos = get_coord(filename)
        
    prefix = (filename.split('.'))[0]
    outfile = prefix+'.xyz'
    with open(outfile, 'w') as f:
        f.write(str(len(pos)))
        f.write('\nxyz from '+filename+' on '+str(today)+' using esta')
        for j in range(len(pos)):
            #f.write("\n"+str(label[j])+"\t"+str("\t".join(map(str, pos[j]))))
            f.write("\n"+str.ljust(label[j],3)+"  "+str("  ".join(["{0:16.9f}".format(i) for i in pos[j]])))

    print('writing data in xyz format from '+ str(filename))
