[docs]    def get_atomic_positions(self):
        '''
        not called in case of ITR list.log ..........
        
        get atomic positiions and atomic labels from  *_P[0-nproc].log  file of GRRM17
                
        '''
        filename = self.filename
        with open(filename, 'r') as f:
            lines=f.readlines()
            lpos = False
            for i in range(len(lines)):
                if "Global minimum = EQ" in lines[i]:
                    lpos = True
                    natoms=0
                    while len((lines[i+natoms+1]).split()) == 4:
                        #print(lines[i+natoms+1])
                        natoms += 1
                    print('no of atoms are: {}'.format(natoms))

                    pos = np.zeros(natoms*3).reshape(natoms,3)
                    iat=0
                    label = []
                    while len((lines[i+iat+1]).split()) == 4:
                        pos[iat:,0] = float((lines[i+iat+1].split())[1])
                        pos[iat:,1] = float((lines[i+iat+1].split())[2])
                        pos[iat:,2] = float((lines[i+iat+1].split())[3])
                        label.append(lines[i+iat+1].split()[0])
                        iat += 1
                        #print(iat)
                    #print(pos)
                    #print(label)
            if lpos:
                return (pos, label, "atm corrdinates in Angstrom")
            else:
                print('\n------------------------------------')
                print("Serious error reading in GRRM file ...")
                print('------------------------------------\n')
                return None



[docs]    def get_all_atomic_positions_from_ITR_list(self):
        '''
        get xyz files: atomic positions and atomic labels from "*_ITR_list.log" file of GRRM17
        '''
        
        filename = self.filename
        with open(filename, 'r') as f:
            lines=f.readlines()
            
            nstructures = 0
            for i in range(0, len(lines)):
                if "# ITR." in lines[i]:
                    nstructures += 1
            print('Number of ITR structures found: {0}'.format(nstructures))

            lpos = False
            #natoms=0
            for i in range(len(lines)):
                #print('natoms:1--', natoms)
                if "# ITR." in lines[i]:
                    lpos = True
                    
                    natoms=0
                    #print('natoms:2--', natoms)
                    while len((lines[i+natoms+1]).split()) == 4 and len((lines[i+natoms+1]).split()[0]) != 12:
                        natoms += 1
                    #print('natoms:3--', natoms)
            print('no of atoms are: {}'.format(natoms))
            
            istr = 0
            pos_all = np.zeros(natoms*3*nstructures).reshape(nstructures,natoms,3) 
            for i in range(len(lines)):
                #print('istr: {}'.format(istr))
                if "# ITR." in lines[i]:
                    pos = np.zeros(natoms*3).reshape(natoms,3)
                    iat=0
                    label = []
                    #while len((lines[i+iat+1]).split()) == 4:
                    while len((lines[i+iat+1]).split()) == 4 and len((lines[i+iat+1]).split()[0]) != 12 :
                        pos[iat:,0] = float((lines[i+iat+1].split())[1])
                        pos[iat:,1] = float((lines[i+iat+1].split())[2])
                        pos[iat:,2] = float((lines[i+iat+1].split())[3])
                        #
                        #EXTREMELY IMPORTANT: label list is destroyed everytime
                        #                     a match in IF statement
                        #                     is found; only last label list is 
                        #                     left and therefore used 
                        #                     for labeling the atoms
                        label.append(lines[i+iat+1].split()[0])
                        #
                        pos_all[istr, iat:, 0] = pos[iat:, 0]
                        pos_all[istr, iat:, 1] = pos[iat:, 1]
                        pos_all[istr, iat:, 2] = pos[iat:, 2]
                        iat  += 1
                    istr += 1
                    #print('testing: MUST be equal')
                    #print('natoms =', natoms,'=','iat=',iat)
            #print('nstructures =', nstructures,'=','istr=',istr)

            if lpos:
                return (pos_all, label, "atm corrdinates in Angstrom")
            else:
                import sys
                print('\n------------------------------------')
                print("Serious error reading in *_ITR_list.log GRRM file ...")
                print('------------------------------------\n')
                sys.exit()
                return None

[docs]    def get_all_xyz_in_a_file(self, comment = None):    
        '''
        get all xyz file by reading the atomic positions and atomic labels: 

        return:
                `outfile`: write data in a single xyz file specified by name `outfile`
                         here outfile  =  prefix+"_"+'all_'+'.xyz' 
        '''
        from datetime import date

        prefix = (self.filename.split('.'))[0]
        today = date.today()

        #pos, label = self.get_all_atomic_positions_from_EQ_list()[0], self.get_all_atomic_positions_from_EQ_list()[1]
        pos, label = self.get_all_atomic_positions_from_ITR_list()[0], self.get_all_atomic_positions_from_ITR_list()[1]        
        print('File'+'-xyz-'*10)
        outfile = prefix+"_"+'all'+'.xyz'    
        with open(outfile, 'w') as f:
            for ipos in range(np.shape(pos)[0]):
                '''
                loop over all atomic postions; ipos
                '''

                f.write(str(len(pos[ipos])))
                
                if comment == None:
                    f.write('\nxyz file created from grrm EQ_list.log on '+str(today)+' using esta')
                elif comment == True:
                    f.write('\nxyz file created from grrm EQ_list.log on '+str(today)+' using esta')
                else:  # Comment == False
                    f.write('\n') # leave the comment line blank as some GUI software doesnt view xyz file
                
                for j in range(np.shape(pos)[1]):
                    f.write("\n"+str.ljust(label[j],3)+"  "+str("  ".join(["{0:16.9f}".format(i) for i in pos[ipos][j]])))
                f.write('\n')
        print('writing all data (atomic structures:{}) in xyz format in a single file'.format(np.shape(pos)[0]))

        '''
        get Energy from *_ITR_list.log file of all TS structures
        '''
        with open(self.filename, 'r') as f:
            lines=f.readlines()
            
            energy = []
            strid = -1
            for i in range(0, len(lines)):
                
                #test -----------------------
                #print('i : {}'.format(i))
                #value = len(lines[i].split()) 
                #print ( value, type(value) )
                #----------------------------
                
                #------------------------------------------------------------------
                # Note: 2 if conditions are used: 
                # warning: if these two keywords are  MUST not be present together 
                # in the *log file
                #-------------------------------------------------------------------
                if "ENERGY  " in lines[i] and len(lines[i].split()) > 3 :
                    print("ENERGY keyword present!!")
                    #energy.append(float(lines[i].split()[1]))
                    #if str(lines[i].split()[1]) != '=':
                    if "=" not in lines[i]:
                        energy.append(float(lines[i].split()[1])*cf.har2kjmolii)
                        strid += 1
                    
                #for new version of grrm; Prof Andrey's log files like this
                if "Relat.Energy               =" in lines[i]:
                    energy.append(float(lines[i].split()[2]))

                # this else MUST not be used otherwise you will never get any energies data from above if
                # statement... I got it .. ha ha ha!!
#                 else: 
#                     print("energy data not accessible .. exiting!!")
#                     import sys
#                     sys.exit()

            str_count = [i for i in range(strid+1)] # str no starting from 0 not 1
            np.savetxt('energy_kj.dat', np.array([str_count, energy-energy[0]]).T)
            np.savetxt('energy_abs.dat', np.array([str_count, energy]).T)
            print('saving energy_kj.dat file containg str no and energy (KJ) wrt initial structure !!')
        return energy-energy[0], energy
