[docs]def get_irc_path(filename, path_direction = False):
    '''
    read log file or any other file containing IRC path information:
    
    input:
    ------
        string; filename
        string;optinal, path_direction can be forward or reversed
                    default is forward direction
        
    output:
    ------
        plot the path in png file named filename.png



     Reading starts from  following ..............
         Energy profile along IRC
         Length (A amu1/2)       Energy (Hartree)
         -61.2536469474         -140.1040646722

         ..
         ..
         ..
         30.6359900582         -140.0788979553
         30.8934801390         -140.0788985106


         Reverse
         Length (A amu1/2)       Energy (Hartree)
         -30.8934801390         -140.0788985106
         -30.6359900582         -140.0788979553
         ..
         ..
         60.7900934845         -140.1040632542
         61.2536469474         -140.1040646722



    '''
    import numpy as np
    import esta.measure.conversion_factors as cf

    
    with open(filename, 'r') as f:
        lines = f.readlines()
        #print(np.shape(lines))
        #print(np.shape(lines[10].split( " ")))
        #print(lines[10].split(" "))
        #print(len(lines))
        
        
        for i in range(len(lines)):
            if "Energy profile along IRC" in lines[i]:
                print('-------IRC path info present------')
                IRCpath=True
                data  = []
                while IRCpath: 
                    j = i + 2              # 2nd lines after the "ENergy progile alogng IRC"
                    data.append (lines[j]) # 2nd lines after the "ENergy progile alogng IRC"
                    if 'Reverse' in lines[j+1]:
                        #i += 1
                        IRCpath = False
                    else:
                        i += 1
                        IRCpath = True
                #print(IRCpath)
                #print(np.shape(data[0].split()))  # tab and space separation .. all together
                #print('data[0:10]:', data[0:10])
                
                # get data for the forward IRC path -----------------------------------
                x_forward = np.zeros(len(data))
                y_forward = np.zeros_like(x_forward)
                for i, ii in enumerate(data):
                    x_forward[i], y_forward[i] = ii.split()[0], ii.split()[1]
                
                print(x_forward, y_forward)
                ndata_pts = len(x_forward)
                #print('data points in forward irc path: {}'.format(ndata_pts))
                
            if "Reverse" in lines[i]:
                print('-----IRC path reverse info present------')
                print('number of data pts in this path Must be: {}'.format(ndata_pts))
                x_reverse = np.zeros(ndata_pts)
                y_reverse = np.zeros_like(x_reverse)
                icounter = 0
                for j in range(ndata_pts):
                    k = j +  2  # data starts from 3rd line after the match "Reverse" => 2 units added
                    x_reverse[j] = lines[i+k].split()[0]
                    y_reverse[j] = lines[i+k].split()[1]
                    icounter += 1
                if ndata_pts == icounter:
                    print('path data pts are equal in forward and reverse IRC path: {}'.format(icounter))
    
        fac = cf.har2kjmolii                    # change y axis units to kJ/mol from hartree/mol
        return (x_forward, y_forward*fac), (x_reverse, y_reverse*fac)

                
        
        

[docs]def get_EQ_IRC(filename, natoms=None):
    '''
    get the eq structures from the IRC calculations 
    
    input: 
    ------ 
            filename: str, name of the IRC log file

    output:
    ------- 
           position: array of shape (2, natoms, 3) ; 2 for two EQ strs 
           label: list of atomic symbols of shape (2, natoms); 2 for eq strs
    '''
    
    import numpy as np
    
    print('Finding: EQ structures from forward as well backward following from IRC calc!!')
    
    
    if natoms is None:
        natoms = get_natoms_IRC(filename)
    
    
    position = np.zeros((2, natoms, 3))
    label = []
    
    
    with open(filename, 'r') as f:
        lines = f.readlines()
        
        for i in range(len(lines)):
            if "IRC FOLLOWING (FORWARD) STARTING FROM FIRST-ORDER SADDLE" in lines[i]:
                print('First order saddle point found --> IRC following in forward direction ..')
                
                ii = i + 3
                j = 0
                steps_forward = []
                while "# STEP" in lines[ii+j]:
                    #print('STEP')
                    steps_forward.append('STEP')
                    j = j + natoms + 4
        print('no. of STEPs in IRC Following (Forward): {}'.format(len(steps_forward)))
                    
            
        ifound = 0            
        for i in range(len(lines)):
            if "Optimized structure," in lines[i]:
                ifound = ifound + 1
                
                if ifound == 1:
                    print('Optimized structure found .. probably for IRC FOLLOWING (FORWARD)!!')
                    label_dummy1 = []
                    for iat in range(1, natoms+1):
                        position[0, iat-1, 0] = float (lines[i+iat].split()[1])
                        position[0, iat-1, 1] = float (lines[i+iat].split()[2])
                        position[0, iat-1, 2] = float (lines[i+iat].split()[3])
                        label_dummy1.append(lines[i+iat].split()[0])
                if ifound == 2:
                    print('Optimized structure found .. probably for IRC FOLLOWING (BACKWARD)!!')
                    #IRC FOLLOWING (BACKWARD) STARTING FROM FIRST-ORDER SADDLE
                    label_dummy2 = []
                    for iat in range(1, natoms+1):
                        position[1, iat-1, 0] = float (lines[i+iat].split()[1])
                        position[1, iat-1, 1] = float (lines[i+iat].split()[2])
                        position[1, iat-1, 2] = float (lines[i+iat].split()[3])
                        label_dummy2.append(lines[i+iat].split()[0])
                    
        label.append(label_dummy1)
        label.append(label_dummy2)
    
    return position, label

    ------- 
           position: array of shape (3, natoms, 3) ; 3 for two EQ and 1 TS strs 
           label: list of atomic symbols of shape (3, natoms); 3 for 2 eq strs and 1 TS strs
    '''
    
    import numpy as np
    
    print('Finding: EQ and TS structures from forward as well backward following from IRC calc!!')
    
    
    if natoms is None:
        natoms = get_natoms_IRC(filename)
    
    
    
    
    with open(filename, 'r') as f:
        lines = f.readlines()

        ifound = 0            
        for i in range(len(lines)):
            if "Geometry (Origin = Center of Mass" in lines[i]:
                ifound = ifound + 1

    print('no. of Geometries found: {}'.format(ifound))

    #position = np.zeros((4, natoms, 3))
    position = np.zeros((int(ifound), natoms, 3))
    label = []

    
    
    
    with open(filename, 'r') as f:
        lines = f.readlines()

        ifound = 0            
        for i in range(len(lines)):
            if "Geometry (Origin = Center of Mass" in lines[i]:
                ifound = ifound + 1
                
                if ifound == 1:
                    print('found Geometry .. probably TS str!!')
                    label_dummy1 = []
                    for iat in range(1, natoms+1):
                        position[0, iat-1, 0] = float (lines[i+iat].split()[1])
                        position[0, iat-1, 1] = float (lines[i+iat].split()[2])
                        position[0, iat-1, 2] = float (lines[i+iat].split()[3])
                        label_dummy1.append(lines[i+iat].split()[0])
                        
                if ifound == 2:
                    print('found Geometry .. probably 1st EQ str!!')
                    label_dummy2 = []
                    for iat in range(1, natoms+1):
                        position[1, iat-1, 0] = float (lines[i+iat].split()[1])
                        position[1, iat-1, 1] = float (lines[i+iat].split()[2])
                        position[1, iat-1, 2] = float (lines[i+iat].split()[3])
                        label_dummy2.append(lines[i+iat].split()[0])

                if ifound == 3:
                    print('found Geometry .. probably 2st EQ str!!')
                    label_dummy3 = []
                    for iat in range(1, natoms+1):
                        position[2, iat-1, 0] = float (lines[i+iat].split()[1])
                        position[2, iat-1, 1] = float (lines[i+iat].split()[2])
                        position[2, iat-1, 2] = float (lines[i+iat].split()[3])
                        label_dummy3.append(lines[i+iat].split()[0])

                #else:
                if ifound == 4:
                    print('found Geometry .. probably 3rd EQ str!! ... ifound value: {}'.format(ifound))
                    label_dummy4 = []
                    for iat in range(1, natoms+1):
                        position[3, iat-1, 0] = float (lines[i+iat].split()[1])
                        position[3, iat-1, 1] = float (lines[i+iat].split()[2])
                        position[3, iat-1, 2] = float (lines[i+iat].split()[3])
                        label_dummy4.append(lines[i+iat].split()[0])
                        
                        
                        
        label.append(label_dummy1)
        label.append(label_dummy2)
        if ifound == 3:
            label.append(label_dummy3)
        if ifound == 4:
            label.append(label_dummy3)
            label.append(label_dummy4)
    
    return position, label

    of IRC calculations
    
    
    .. Check following matches::
       Thermochemistry at  298.150 K,    1.000 Atm

       Thermochemistry (after the above replacements)
    """
    
    with open(filename, 'r') as f:
        lines = f.readlines()

        ifound = 0            
        for i in range(len(lines)):
            if "Geometry (Origin = Center of Mass" in lines[i]:
                ifound = ifound + 1
        
    print('no. of free energy values: {}'.format(ifound))
    
    free_energy = np.zeros((int(ifound)))
    #print(free_energy)
    
    
    with open(filename, 'r') as f:
        lines = f.readlines()

            if "Thermochemistry at" in lines[i]:
                print('found: Thermochemistry at')

                # see 14 line for free energy

                print('j: {}; free_energy[j]: {}'.format(j, free_energy[j]))

                print('found: Thermochemistry (use')
