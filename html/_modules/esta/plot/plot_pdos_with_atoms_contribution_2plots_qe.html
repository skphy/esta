[docs]def get_pdos_files(atm_info):
    '''
    
    extract files based on the atomic number of atoms
    
    input: 
    ------
        atm_info: list of integers for atoms whose pdos is needed with all s,p,d...charaters
    
    output:
    ------
        fileslist: list of all files whose pdos is needed
    '''

    import glob
    #atm_info=[128, 129]
    print('atom info: {}'.format(atm_info))
    fileslist = []
    for iat in atm_info:
        fl = glob.glob('*atm#'+str(iat)+'('+'*')
        #print(fl)
        for i in fl:
            fileslist.append(i)
    #print(fileslist)
    return fileslist




[docs]def get_range_int_from_intervals(list_of_intervals):
    '''
    get range of integers from intervals (in the form of strings)
    
    input:
    ------
        list of intervals: e.g. ['1','2-4','6-10'] 
        note that elements of list are strings
        
    output:
    ------
        range of intervals: e.g. [1, 2, 3, 4, 6, 7, 8, 9, 10]
        note that elements are integers
    '''

    #listt = ['1','2-4','6-10']
    ele = []
    for i in list_of_intervals:
        if len(str(i).split("-")) == 2:
            val = str(i).split("-")
            val_ini = int(val[0])
            val_fin = int(val[1])
            #print(val_ini, val_fin)
            for j in range(val_ini , val_fin+1 ):
               ele.append(j) 
        else: 
            ele.append(i)
    elem = [int(e) for e in ele]
    return elem

    '''
    - summing the 2nd, and 3rd columns of the following multiple files and 
      create new file with 0th column (from one file), 1st summed and 2nd summed 
      columns from all files
      
    input:
    ------
        filelist: list of all files whose columns hve to be processed
        
    output:
    ------
         sum_data_file: file with resultant columns,
         data_array: array[:,:,0:3] data with processed columns
        
    '''

    import sys
    import numpy as np

#     filelist=['alk.pdos_atm#128(C)_wfc#2(p)', \
#               'alk.pdos_atm#129(C)_wfc#2(p)', \
#               'alk.pdos_atm#129(C)_wfc#2(p)' ]
    # filelist=['d1.dat', 'd2.dat', 'd3.dat' ]

    rows = []
    for ifile in filelist:
        with open(ifile,'r') as f1:
            lines = f1.readlines()
            rows.append(np.shape(lines)[0])

    print('no. of rows in all files: {}'.format(rows))
    result = checkEqual_list(rows)
    if result:
        print('no. of rows are consitent in all the files.')
    else:
        print('no. of rows are not consitent in all the files.')
        print('exiting ..')
        sys.exit()

    columns = 3                   # 3 columns; energy, ldos_up, ldos_dwn
    total_files = len(filelist)+1 # 1 more file for adding dos 
    nrows = rows[0]-1             # 1 less as 1st line is comment line
    data_array = np.zeros(( total_files, nrows, columns)) 
    # print(np.shape(data_array))

    for i, ifile in enumerate(filelist):
        with open(ifile,'r') as f1:
            lines = f1.readlines()
            content = [ x.strip() for x in lines ]
            for icon in range(len(content)):
                #print('icon: ', icon)
                if icon > 0:  # icon = 0 is comment line
                    data_array[i, icon-1, 0] = float(content[icon].split()[0])
                    data_array[i, icon-1, 1] = float(content[icon].split()[1])
                    data_array[i, icon-1, 2] = float(content[icon].split()[2])

    print('data stored in the data_array from all files')                
    print('data_array: data_array[0,0:10,0:1]')                    
    #print(data_array[:,:,:])                               
    print(data_array[0,0:10,0:2])                               
    # sys.exit()    

    # adding dos columns i.e 2nd and 3rd columns of the input files                    
    array_sum = np.sum(data_array[:,:,1:], axis=0)
    data_array[-1,:,0] = data_array[0,:,0]
    data_array[-1,:,1:] = array_sum
    data_sum = data_array[-1,:,:]


    # # sum_data_file = filelist[0].split(".")[0]
    # # alk.pdos_atm#102(N)_wfc#1(s)
    # # print(filelist[0].split("#")[1].split("(")[0]) 
    # atm='A'+str(filelist[0].split("#")[1].split("(")[0])
    # # print(filelist[0].split("#")[2].split("(")[0])
    # orb = '-S'+str(filelist[0].split("#")[2].split("(")[0])
    # sum_data_file = str(atm)+str(orb)+'_sum.dat'

    namefile = []
    for ifile in range(len(filelist)):
        atm = 'A'+str(filelist[ifile].split("#")[1].split("(")[0])
        orb = '-S'+str(filelist[ifile].split("#")[2].split("(")[0])
        namefile.append(str(atm)+str(orb)+'_')

    print('namefile: {}'.format(namefile))
    sum_data_fil = "".join(namefile)
    sum_data_file = str(sum_data_fil)+'sum.dat'
    np.savetxt(sum_data_file, data_sum)

    # sys.exit()

    ltest = False
    if ltest:
        print('--summed data ------')
        array_sum = np.sum(data_array[:,:,1:], axis=0)
        print(array_sum[0:10])

        print('--initial data --')
        print(0,data_array[0,0:10, 1:])
        print(1,data_array[1,0:10, 1:])
        print(2,data_array[2,0:10, 1:])

        #------
        print('result final data')
        data_array[-1,:,0] = data_array[0,:,0]
        data_array[-1,:,1:] = array_sum
        print(data_array[:,0:10,:])
    
    return sum_data_file, data_array
