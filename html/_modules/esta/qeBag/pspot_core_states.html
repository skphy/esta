[docs]def id_orb2_indx(id_orb):
    '''
    change atomic orbital id name to id number
    
    input:
        string: e.g. S, P, D, and F
    output:
        scalar: e.g  0, 1, 2, and 3
    '''
    
    if id_orb == 'S':
        return 0
    if id_orb == 'P':
        return 1
    if id_orb == 'D':
        return 2
    if id_orb == 'F':
        return 3

     Input:
     -----
         string: UPF file for a specific atom; e.g: Ni.pbe-n-rrkjus_psl.1.0.0.UPF

    for child in root:
        print('tag:', child.tag)
        print('attrib:', child.attrib)
        
    print ('all the elements in entire tree:')
    print ([elem.tag for elem in root.iter()])
    print('\n')
    #print([elem.attrib for elem in root.iter()])

    
    for el1 in root.findall('PP_MESH'):

        mesh_value = el1.attrib['mesh']

        el2 = el1.find('PP_R')
    #testing:
    rmesh = el2.text.split()
    print('mesh type:',type(rmesh))

    rrmesh = np.asarray(rmesh)
    #print(rrmesh, type(rrmesh), np.shape(rrmesh))
    np.savetxt('rmesh.dat', rrmesh, fmt='%s')
    
    
    for el1 in root.findall('PP_GIPAW'):
        #print(el1)
        #print(type(el1))
        gipaw_format = el1.attrib['gipaw_data_format']
        el2 = el1.find('PP_GIPAW_CORE_ORBITALS')

        xcore_orbitals = np.zeros((core_orbitals, len(rrmesh)))
