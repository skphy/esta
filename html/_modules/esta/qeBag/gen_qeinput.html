[docs]def min_scalarlist(sequence_list):
    """return the minimum element from scalar elements of a list
    """
    lowest = sequence_list[0] 
    for i in sequence_list:
        if i < lowest:
            lowest = i
    return lowest

[docs]def get_N1_N2_N3_in_MP_grid(posfile = None, location = None, l_metal_insulator = None):
    '''    
        For auto-matic k-grid(taken from vasp manual): 

        N 1 = max(1, l ∗ | b 1 | + 0.5)
        N 2 = max(1, l ∗ | b 2 | + 0.5)
        N 3 = max(1, l ∗ | b 3 | + 0.5).
        Symmetry is used to map equivalent k-points to each other, 
        which can reduce the total number of k-points significantly. 
        Usefulvalues for the length (l) vary between 10 (large gap insulators) 
        and 100 (d-metals).
        
        NOTE:
            If length of lv1, lv2, lv3 > 2; then taken N1/N2/N3 = 1, acoordingly
        
        NOTE: need testing of this method/functions????????

    '''
    import numpy as np
    import esta
    from esta.vaspBag.inout.crystal_lattice import CrystalLattice as CLattice

    # How to get the reciprocal lattice vector from CLattice class 
    #pos_obj = CLattice('POSCAR', './')
    #pos_obj.get_poscar()
    
    if posfile is None:
        posfile = 'POSCAR'
    if location is None:
        location = './'
        
    clatt = CLattice(posfile, location)
    positions, vol, lv1, lv2, lv3, atmsymbol, natmtyp, natoms = clatt.get_poscar()
    
    # print(pos_obj.reciprocal_lattice())
    b1, b2, b3 = clatt.reciprocal_lattice()
    #print(b1, b2, b3)
    #print(np.linalg.norm(b1), np.linalg.norm(b2), np.linalg.norm(b3))

    # need a better way to find following parameters: SK
    l_metal = 100
    l_insulator = 10
    l_optimal = 30 #50
    
        
    if l_metal_insulator == None:
        N1 = np.rint(np.maximum(1, l_optimal * np.abs(np.linalg.norm(b1)/np.pi/2) + 0.5))
        N2 = np.rint(np.maximum(1, l_optimal * np.abs(np.linalg.norm(b2)/np.pi/2) + 0.5))
        N3 = np.rint(np.maximum(1, l_optimal * np.abs(np.linalg.norm(b3)/np.pi/2) + 0.5))
    if l_metal_insulator == 'metal':
        N1 = np.rint(np.maximum(1, l_metal * np.abs(np.linalg.norm(b1)) + 0.5))
        N2 = np.rint(np.maximum(1, l_metal * np.abs(np.linalg.norm(b2)) + 0.5))
        N3 = np.rint(np.maximum(1, l_metal * np.abs(np.linalg.norm(b3)) + 0.5))
    if l_metal_insulator == 'insulator':
        N1 = np.rint(np.maximum(1, l_insulator * np.abs(np.linalg.norm(b1)) + 0.5))
        N2 = np.rint(np.maximum(1, l_insulator * np.abs(np.linalg.norm(b2)) + 0.5))
        N3 = np.rint(np.maximum(1, l_insulator * np.abs(np.linalg.norm(b3)) + 0.5))

    length_cutoff = 20. # Angstom
    print('\nReal space length cutoff (angstrom): {}'.format(length_cutoff))
    if np.linalg.norm(lv1) >= length_cutoff:
        N1 = 1
    if np.linalg.norm(lv2) >= length_cutoff:
        N2 = 1
    if np.linalg.norm(lv3) >= length_cutoff:
        N3 = 1
    print('N1 N2 N3: {} x {} x {}\n'.format(int(N1), int(N2), int(N3)))
        
    return int(N1), int(N2), int(N3)
