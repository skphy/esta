[docs]def min_scalarlist(sequence_list):
    """return the minimum element from scalar elements of a list
    """
    lowest = sequence_list[0] 
    for i in sequence_list:
        if i < lowest:
            lowest = i
    return lowest



[docs]def max_scalarlist(sequence_list):
    """return the max element from scalar elements of a list
    """
    highest = sequence_list[0] 
    for i in sequence_list:
        if i > highest:
            highest = i
    return highest

[docs]def get_cutoff_wfc_rho(loc, ps_info, atmsymbol, lcopy_pseudo=True):
    '''
    find 
    - the min kinetic energy cutoff
    - the min charge density cutoff by the reading the PP header files
    
    input:
        string: loc: location of PP files
        string: ps_info: PP information
        list: atmsymbols: atomic labels
        boolen: lcopy_pseudo: whether to copy PP to current dir from PP database
        
    output:
        scalar: min kinetic enrgy cutoff
        scalar: min charge density cutoff
    '''

    import numpy as np
    #from mendeleev import element
    from glob import glob
    import os
    from shutil import copy
    
    pseudo_list=[]
    for ii, i in enumerate(atmsymbol):
        #istt = glob(loc+str(i)+'.pbe*kjpaw*')
        listt = glob(loc+str(i)+ps_info)
        #print(listt)
        plist=[]
        print('pseudo-potential information:')
        for jj in listt:
            print(jj.split('/')[-1])
            dum=jj.split('/')[-1]
            plist.append(dum)
        pseudo_list.append(plist)
        string = "  ".join(k for k in pseudo_list[ii])
        #f.write(str(i)+'\t'+str(element(i).mass)+'\t'+str(pseudo_list1[ii])+nln)
        #f.write(str(i)+'\t'+str(element(i).mass)+'\t'+string+nln)  
    
    
        # storing 1 PP for each element; not more than 1 PPs
        pseudo_list_1for1 = []
        for i in pseudo_list:
            pseudo_list_1for1.append(i[0])
        #print(pseudo_list_1for1)    
        
        #from shutil import copyfile
        #copyfile(loc, dst) # loc and dst are dir here
        #from shutil import copy
        for ifile in pseudo_list_1for1:
            copy(loc+'/'+str(ifile), './')
        
        #find kinetic energy cutoff and charge density cutoff from header of PP files
        kinetic_energy = []
        charge_density = []
        for ifile in pseudo_list_1for1:
            with open(ifile, 'r') as filee:
                lines = filee.readlines()
                for ilines in lines:
                    if "Suggested minimum cutoff for wavefunctions:" in ilines:
                        #print('minimum cutoff for wave-function (Ry):', ilines.split()[5])
                        kinetic_energy.append(float(ilines.split()[5]))
                    if "Suggested minimum cutoff for charge density:" in ilines:
                        #print('minimum cutoff for charge density (Ry):', ilines.split()[6])
                        charge_density.append(float(ilines.split()[6]))
        #print('kinetic energy: {} '.format(*kinetic_energy))
        #print('charge density: {} '.format(*charge_density))
        
        #print('kinetic energy:')
        #print(*kinetic_energy)
        #print('charge density:')
        #print(*charge_density)
        
        kinetic_energy_cutoff = max_scalarlist(kinetic_energy)
        charge_density_cutoff = max_scalarlist(charge_density)
        #print('suggested kinetic energy cutoff (Ry): {}'.format(kinetic_energy_cutoff))
        #print('suggested charge density cutoff (Ry): {}'.format(charge_density_cutoff))

    return kinetic_energy_cutoff, charge_density_cutoff
