[docs]def energies_kpts(filename):    
    '''
    finding:
    -nbands, band-energies
    -nkpts, kpt-coordinate 
    -band gap for spin up channel
    -band gap for spin down channel
    -band gap between valence band(of spin up/down) and cond. band (spin up/down)
    for unpolarised/polarised qe scf.out file

    NOTE: use verbosity = 'high'  to print the band energies

    
    input: 
        name of scf.out file with verbosity high to be used in qe pw.x 
        calculation
        
        
        

    '''
    print()
    print('WARNING: TESTED FOR NON-SPIN POLARISED CALCULATIONS. NOT TESTED FOR SPIN POLARISED CALCULATION')
    print('WARNING: TESTED FOR NON-SPIN POLARISED CALCULATIONS. NOT TESTED FOR SPIN POLARISED CALCULATION')
    print('')
    import numpy as np
    with open(filename, 'r') as f:
        lspin = False
        lines = f.readlines()
        for i in range(len(lines)):
            if "number of Kohn-Sham states=" in lines[i]:
                nband = int(lines[i].split()[4])
            if "number of k points=" in lines[i]:
                nkpt = int(lines[i].split()[4])
            if 'the Fermi energy is' in lines[i]:
                #print('Fermi level',lines[i].split()[4])
                efermi = float(lines[i].split()[4])
            if "SPIN UP" in lines[i]:  # or SPIN DOWN
                lspin = True
        print('no of bands: {}, no of kpts: {}, Fermi level: {}'.format(nband, nkpt, efermi))

        energies = np.zeros((nkpt, nband, 2)) # 2 for spin polarised calc
        occupations = np.zeros_like(energies)
        coord_kpts =[]
        for i in range(len(lines)):
            if "End of self-consistent calculation" in lines[i]:
                iskip = 0
                iiskip=0
                for ispin in range(2):
                    for ik in range(nkpt):
                        energy = []
                        occupation = []
                        
                        if lspin:
#                             lines[i+1 + iskip]
#                             lines[i+2 + iskip] # spin info (for iskip=0)
#                             lines[i+3 + iskip]
                            shift = 3
                        else:
                            shift = 0
                        
#                         lines[i+4 + iskip]
#                         lines[i+5 + iskip] # k point info for iskip=0
#                         lines[i+6 + iskip]

                        kdata_x=((lines[i+2+shift+iskip+iiskip]).split())[2] #-this is kpt-coord x compont
                        kdata_y=((lines[i+2+shift+iskip+iiskip]).split())[3] #-this is kpt-coord y compont
                        kdata_z=((lines[i+2+shift+iskip+iiskip]).split())[4] #-this is kpt-coord z compont
                        kdata =[kdata_x, kdata_y, kdata_z]
                        coord_kpts.append(kdata)

                        for ii in lines[i+4+shift+iskip+iiskip].split(): # band energy line 
                            energy.append(float(ii))

                        icount=1
                        while len(energy) < nband:
                            data=(lines[i+4+shift+iskip+icount+iiskip]).split()
                            for idata in data:
                                energy.append(float(idata))
                            icount += 1
                        energies[ik,:,ispin] = [energy[i] for i in range(nband)]
                        
                        
                        #======occupation related part===========================
                        ioccupation_shift = 2
                        ioccupation_lines = icount
                        
                        for ii in lines[i+4+shift+iskip+iiskip+ioccupation_shift+ioccupation_lines].split(): # band energy line 
                            occupation.append(float(ii))

                        jcount=1
                        while len(occupation) < nband:
                            occ_data=(lines[i+4+shift+iskip+iiskip+ioccupation_shift+ioccupation_lines+jcount]).split()
                            for idata in occ_data:
                                occupation.append(float(idata))
                            jcount += 1
                        occupations[ik,:,ispin] = [occupation[i] for i in range(nband)]
                        #==========================================================
                        
                        
                        #iskip += 3 + icount
                        iskip += 3 + icount + ioccupation_shift + ioccupation_lines
                        
                    iiskip = iiskip + 3 
                    if not lspin:
                        break
        
        # scale energies wrt Fermi level
        energies = energies- efermi            
        
        #=================================================
        def min_of_positive_values(energies_array):
            return min(v for v in energies_array if v > 0)
        def max_of_negative_values(energies_array):
            return max(v for v in energies_array if v < 0)
        #=================================================
        
        #for spin up
        VB_values = []
        CB_values = []
        
        #for spin down
        VB_values1 = []
        CB_values1 = []

        for ik in range(nkpt):
            e = energies[ik,:,0]
            VBm = max_of_negative_values(e)
            VB_values.append(VBm)    
            CBM = min_of_positive_values(e)
            CB_values.append(CBM)
            
            if lspin:
                e1 = energies[ik,:,1]
                VBm1 = max_of_negative_values(e1)
                VB_values1.append(VBm1)    
                CBM1 = min_of_positive_values(e1)
                CB_values1.append(CBM1)
            
        #for spin up
        #print('VB_values', VB_values)
        #print('CB_values', CB_values)
        band_gap = np.max(VB_values) - np.min(CB_values)
        idx_VB_kpt = np.argmax(VB_values)
        idx_CB_kpt = np.argmin(CB_values)

        lkpt_info = False
        if lkpt_info:
            print('VB_values:')
            print(VB_values)
            print(np.max(VB_values))
            print('idx_VB_kpt')
            print(idx_VB_kpt)
            print(coord_kpts[idx_VB_kpt])
            print(occupations[idx_VB_kpt])
        
            print('CB_values:')
            print(CB_values)
            print(np.min(CB_values))
            print('idx_CB_kpt')
            print(idx_CB_kpt)
            print(coord_kpts[idx_CB_kpt])
            print(occupations[idx_CB_kpt])
        
        band_gap_up = np.abs(band_gap)
        #print('band gap up: {}'.format(band_gap_up))
        
        if lspin:
            #for spin down
            #print('VB_values1', VB_values1)
            #print('CB_values1', CB_values1)
            band_gap1 = np.max(VB_values1) - np.min(CB_values1)

            band_gap_down = np.abs(band_gap1)
            #print('band gap down : {}'.format(band_gap_down))
        

            #band-gap: this is true band gap; can be difference of VBM 
            #          spin up and CBM of spin down channals or so on
            VBm_ = np.max(np.array([np.max(VB_values),np.max(VB_values1)])) 
            CBM_ = np.min(np.array([np.min(CB_values),np.min(CB_values1)])) 
            band_gap = np.abs(VBm_ - CBM_)
    if lspin:
        return energies, coord_kpts, band_gap_up, band_gap_down, band_gap, lspin, occupations
    else:
        return energies, coord_kpts, band_gap_up, None, None, lspin, occupations
