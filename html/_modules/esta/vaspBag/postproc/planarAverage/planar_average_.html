    def planar_average(func3D, nx, ny, nz, direction):
        '''
        planar average along direction given by the direction string
        '''
        print('Planar averaring along direction:', direction)

        if direction == 'x':
            print ('Planar averaring along x direction:')
            _func3D_av = np.zeros(nx) 
            for i in range(nx):
                for j in range(ny):
                    for k in range(nz):
                        _func3D_av[i] += func3D[k,j,i]
                _func3D_av[i] = _func3D_av[i]/(ny*nz)

        if direction == 'y':
            print ('Planar averaring along y direction:')
            _func3D_av = np.zeros(ny)  
            for j in range(ny):
                for i in range(nx):
                    for k in range(nz):
                        _func3D_av[j] += func3D[k,j,i]
                _func3D_av[j] = _func3D_av[j]/(nz*nx)

        if direction == 'z':
            print ('Planar averaring along z direction:')
            _func3D_av = np.zeros(nz)                    
            for k in range(nz):
                for i in range(nx):
                    for j in range(ny):
                        _func3D_av[k] += func3D[k,j,i]
                _func3D_av[k] = _func3D_av[k]/(nx*ny)
        return _func3D_av
