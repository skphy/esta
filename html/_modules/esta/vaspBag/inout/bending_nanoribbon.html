[docs]def get_bent_structure(angle_degree):
    '''
    
    input: 
    ------
            scalar: angle_degree
            
            input file name must be 'POSCAR'
    
    output:
    -------
            string: filename containg data for bent str in POSCAR format
    
    
    
    
    read related information etc from other folder ..for bending

    Date: 2018
    Author: sonu
    Email: sonukumar.physics@gmail.com

    '''
    import numpy as np
    import sys
    import esta
    
    from esta.vasp_bag.inout import read_poscar

    from datetime import date
    today = date.today()
    
    # sys.path.append('../../') # directory containg vasp_bag folder
#     sys.path.insert(0,'/home/sonu/ipython_exercises/vasp_bag')
    #import vasp_bag
#     from inout import read_poscar
    
    out=read_poscar.get_poscar("./")
#     sys.exit()

    # input coordinates must be in cartesian basis
    # print np.shape(read_poscar.tau)

#     in_coord = read_poscar.tau
    #out return tau_cartesian, real_volume, LV1, LV2, LV3
    in_coord = out[0] #sk
    
    print("input coordinates shape:")
    print(np.shape(in_coord))
    print(np.max(in_coord[:,0]))
#     print read_poscar.LV1[0]


    #lattice constant along x direction; direction which will be bended .. 
    # print read_poscar.LV1[0]
#     length_to_bend = read_poscar.LV1[0]
    length_to_bend = out[2][0]  #sk
    
    # length_to_bend = np.max(in_coord[:,0])
    print("length to bend (ang): {0:6.2f}".format(length_to_bend))

    angle_in_degree = angle_degree
    theta = angle_in_degree * np.pi/180.

    if theta == int(0.):
        print("bending by zerp degree not possible.")
        print(sys.exit())

    print("bending angle in radian: {0:6.2f}".format(theta))
    print("bending angle in degree: {0:6.2f}".format(angle_in_degree))

    # radius of circle/arc
    # radius = length_to_bend/2./np.pi
    radius = length_to_bend/theta
    print("radius (ang): {0:6.2f}".format(radius))



    # transformed coordinates
    t_coord = np.zeros_like(in_coord)

    for i in range(len(in_coord)):
        ## *Note: xy plane becomes xz plane; so I did transformation so that plane 
        #         does not change*    ----need to recheck again --
        #
        # Note: bending takes place in x-y plane and arc protrudes in z direction -- procedure for folding
        # Note: xz or ac plane remains same ---checked
        # 
        t_coord[i,0] = (radius + in_coord[i,2]) * np.cos(theta*in_coord[i,0]/length_to_bend)
        t_coord[i,2] = (radius + in_coord[i,2]) * np.sin(theta*in_coord[i,0]/length_to_bend)
        t_coord[i,1] = in_coord[i,1]

    
    lcoord = False
    if lcoord:
        # np.savetxt('coord.dat', np.array([t_coord[:,0], t_coord[:,1], t_coord[:,2]]).T)  
        print("Generating coord.dat file ...")
        np.savetxt('coord.dat', np.array(t_coord))    
       
    print('Generating  '+'degree_'+str(int(angle_in_degree))+'_bentPOSCAR ..')
    with open('degree_'+str(int(angle_in_degree))+'_bentPOSCAR','w') as f:
        f.write('# poscar created by '+str(angle_in_degree)+' deg-bend on '+str(today)+' using esta\n')
        
        f.write("1\n")
        f.write( ' \t '.join(str(i) for i in out[2]) + "\n")
        f.write( ' \t '.join(str(i) for i in out[3]) + "\n")
        f.write( ' \t '.join(str(i) for i in out[4]) + "\n")
        
        f.write( ' \t '.join(map(str,out[5]))   + "\n")
        f.write( ' \t '.join(map(str,out[6])) + "\n")
        
        f.write("Cartesian\n")
        for i in range(out[7]):
            #f.write(" \t".join(map(str, t_coord[i,:])) + "\n")
            f.write(str("".join(["{0:18.12f}".format(i) for i in  t_coord[i,:] ]))+'\n')            


if __name__=='__main__':
    '''
    calling bending method ..
    '''
    angle_degree=60.
    get_bent_structure(angle_degree)
