[docs]    def reciprocal_lattice(self):
        '''
        method to calculate the reciprocal lattice vectors        

        input: real space lattice vectors: LV1, LV2, LV3
        output: reciprocal space lattice vectors: b1, b2, b3 
                - note: a factor of 2pi is included in b1, b2, and b3
        '''
        import numpy as np

        LV1 = self.LV1
        LV2 = self.LV2
        LV3 = self.LV3
        real_volume = np.dot(LV1, np.cross(LV2,LV3))
        b1 = 2*np.pi * np.cross(LV2, LV3)/ real_volume
        b2 = 2*np.pi * np.cross(LV3, LV1)/ real_volume
        b3 = 2*np.pi * np.cross(LV1, LV2)/ real_volume
        
        return b1, b2, b3

            '''

            ulist = []

[docs]    def get_grouped_list(self, llist):
        '''
        get same string elements of list grouped together
        '''

        res = self.get_unique_list(llist)
        grouped_list = []
        idx = []

        for i in range(len(res)):
            for j in range(len(llist)):
                if res[i] == llist[j]:
                    idx.append(j)
        for i in idx:
            grouped_list.append(llist[i])

        return(grouped_list, idx)

        '''
        get number of list entries of each type

        return:

              list of integers for each type of atoms specifying their number
        '''    
        glist = self.get_grouped_list(inp)[0]
        ulist = self.get_unique_list(inp)
        #print(ulist)

        neach = []
        for i in ulist:
            k = 0
            for j in glist:
                if i == j:
                    k += 1
            neach.append(k)
        #print (neach)
        return(neach)
