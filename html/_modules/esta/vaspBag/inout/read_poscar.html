#!/usr/bin/env python2.7
'''
routine to read POSCAR file (VASP)

Input: default POSCAR file is taken from current dir.

Returns:
     if selective dynamics:
         return tau_cartesian, real_volume, LV1, LV2, LV3, atomFix, atomMove
     else:
         return tau_cartesian, real_volume, LV1, LV2, LV3

Note:
        modification done to coordinates if -ve by more than 1. unit of 
        angstrom: see if there is a better way to handle it     
'''

# import copy
import numpy as np


[docs]def get_poscar(location):
    '''
    method to read poscar file
    
    Input: location of "POSCAR" file
    
    returns:
     if selective dynamics:        
        return tau_cartesian, real_volume, LV1, LV2, LV3, atm_type, natom_type, natoms, atomFix, atomMove
     else: 
        return tau_cartesian, real_volume, LV1, LV2, LV3, atm_type, natom_type, natoms
         
    '''
#     print(" ")
#     print( "-------------------------------------------------")
    print(( "importing poscar file from locatioin:", location))
#     print( "-------------------------------------------------")

    with open(location + 'POSCAR') as f:
        print ("")
        lines = f.readlines()

        LV1 = np.zeros(3)
        LV2 = np.zeros(3)
        LV3 = np.zeros(3)
        for i in lines:

            if i == lines[2]:
                _data = lines[2].split()
                LV1[0] = float(_data[0])
                LV1[1] = float(_data[1])
                LV1[2] = float(_data[2])

            if i == lines[3]:
                _data = lines[3].split()
                LV2[0] = float(_data[0])
                LV2[1] = float(_data[1])
                LV2[2] = float(_data[2])

            if i == lines[4]:
                _data = lines[4].split()
                LV3[0] = float(_data[0])
                LV3[1] = float(_data[1])
                LV3[2] = float(_data[2])

            if i == lines[5]:
                atm_type = lines[5].split()
                no_of_type = len(atm_type)
                print(( "no_of_type --> types of atoms:", no_of_type))
                for i in range(len(atm_type)):
                    print(( 'atm_type --> atom type',i,':', atm_type[i]))
            if i == lines[6]:
                _data = lines[6].split()
                # print len(data)
                natom_type = np.zeros(no_of_type, dtype=int)
                natoms = 0
                for k in range(len(_data)):
                    #print _data[k], type(int(_data[k]))
                    natom_type[k] = int(_data[k])
                    natoms += int(_data[k])

    for i in range(len(_data)):
        print(( "natom_type --> no of atoms of type",i,':', natom_type[i]))

    # Selective dynamics
    l_SelectDynamics = False
    #Note: better strip the lines[7]--todo
    if str(lines[7])[0] == 's' or str(lines[7])[0] == 'S':
        print ("")
        print("Selective dynamics")
        l_SelectDynamics = True
        
        coord_type = lines[8] 
        print(('Cartesian/Fractional?:', coord_type))

        tau = (np.zeros(natoms*3)).reshape(natoms, 3)  
        lFT = [] #np.zeros_like(tau)
        with open(location + 'POSCAR') as f:
            print(" ") 
            lines = f.readlines()

            for i in lines:
                if i == lines[9]:
                    for j in range(natoms):
                        _data = lines[9+j].split()
                        tau[j,0] = float(_data[0])
                        tau[j,1] = float(_data[1])
                        tau[j,2] = float(_data[2])
                        #print _data[3:], type(_data[3:3])
                        lFT.append(_data[3:6])  # last index excluded                        
            if natoms != len(lFT):
                print("number of atoms fixed must be equal to no of atoms in POSCAR")

            atomFix = []
            atomMove = []
            for i, counter in enumerate(lFT):
                if lFT[i][0] == 'F' or lFT[i][0] == 'f' or  lFT[i][1] == 'F' or lFT[i][1] == 'f'  or lFT[i][2] == 'F' or lFT[i][2] == 'f':
                    #print("Fixed atoms")
                    atomFix.append(i)
                elif lFT[i][0] == 'T' or lFT[i][0] == 't' or  lFT[i][1] == 'T' or lFT[i][1] == 't'  or lFT[i][2] == 'T' or lFT[i][2] == 't':
                    #print("Atoms to be Moved")
                    atomMove.append(i)
        atomFix = np.array(atomFix)+1
        atomMove = np.array(atomMove)+1
        
    elif str(lines[7])[0] == 'd' or str(lines[7])[0] == 'D' or str(lines[7])[0] == 'C' or str(lines[7])[0] == 'c':
        coord_type = lines[7] 
        print(('Cartesian/Fractional?:', coord_type))
        
        tau = (np.zeros(natoms*3)).reshape(natoms, 3)    
        with open(location + 'POSCAR') as f:
            print(" ") 
            lines = f.readlines()
            for i in lines:
                if i == lines[8]:
                    for j in range(natoms):
                        _data = lines[8+j].split()
                        tau[j,0] = float(_data[0])
                        tau[j,1] = float(_data[1])
                        tau[j,2] = float(_data[2])
#     print ('lattice vectors (Angstrom): LV1, LV2, LV3')
#     print (LV1)
#     print (LV2)
#     print (LV3)
    real_volume = np.dot(LV1, np.cross(LV2,LV3))
#     print ('lattice vectors (Bohr): Lv1, Lv2, Lv3')
    Lv1 = LV1/0.529177249 
    Lv2 = LV2/0.529177249 
    Lv3 = LV3/0.529177249 

    print(('total no of atoms:',natoms))
#     print ('Cartesian/Fractional?:', coord_type)
    # see if better assigment can be done
    atms_position = tau

    # _Cell Matrix: (C style or Python style)
    _Cell_Matri = np.array([LV1,LV2,LV3])
    Cell_Matrix = _Cell_Matri.T
    print ('Cell Matrix: (Fortran style)')
    print (Cell_Matrix)

    if coord_type.strip() == 'Direct' or coord_type.strip() == 'direct':  
       #print ('Coordinates type is DIRECT')
       tau_cartesian = np.zeros(len(tau)*3).reshape(len(tau),3)
       # print tau_cartesian, np.shape(tau_cartesian)
       #print ("Conversion of coordinates from direct to cartesian type:")
       for i in range(len(tau)):
           #     print i, tau[i]
           tau_cartesian[i] = Cell_Matrix.dot(tau[i])  #Matric * vector multiplication
    else:
       tau_cartesian = tau

    is_negative = False
    for i in range(len(tau)):
       if tau_cartesian[i,0] < 0. or tau_cartesian[i,2] < 0. or tau_cartesian[i,1] < 0. :
           print(("WARNING: -ve coordinates ..", tau_cartesian[i,:]))
           is_negative = True

    print("")      
    inv_Cell_Matrix = np.linalg.inv(Cell_Matrix)

    tau_direct = np.zeros(len(tau)*3).reshape(len(tau),3)
    if coord_type.strip() == 'Cartesian' or coord_type.strip() == 'cartesian':  
    #     print 'Converting coordinates from Cartesian to direct:'
        for i in range(len(tau)):
            #print inv_Cell_Matrix.dot(tau[i,:])
            tau_direct[i] = inv_Cell_Matrix.dot(tau[i])  #Matric * vector multiplication
    #         print i, '{:08.6f} {:08.6f} {:08.6f}'.format(float(tau_direct[i][0]),float(tau_direct[i][1]),float(tau_direct[i][2]))
    # #         print i, '{:08.6f} {:08.6f} {:08.6f}'.format(float(tau_direct[i,0]),float(tau_direct[i,1]),float(tau_direct[i,2]))

    # added 19Jan2019
    if l_SelectDynamics:
        return tau_cartesian, real_volume, LV1, LV2, LV3, atm_type, natom_type, natoms, atomFix, atomMove
    else: 
        return tau_cartesian, real_volume, LV1, LV2, LV3, atm_type, natom_type, natoms
