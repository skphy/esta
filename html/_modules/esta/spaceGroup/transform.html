[docs]    def get_supercell(self, scale):
        '''
        - create supercell by shifting ALL atoms in space with scaling 
        [scale1,scale2,scale3]
        
        - looping is performed along three directions of lv's vectors
        
        - loops are:
            i = 0,1,2 ... sclae1
            j = 0,1,2 ... scale2 
            k = 0,1,2 ... scale3
        
        - total atoms in supercell = atoms in unit-cell * np.product([scale1,scale2,scale3])
        
        '''
        import numpy as np
        #print ('creating supercell from input/unit cell')
        
        scell = np.zeros((3,3))
        scell[0,:] = scale[0] * self.lv1  
        scell[1,:] = scale[1] * self.lv2  
        scell[2,:] = scale[2] * self.lv3  
        #print(self.lv1, self.lv2, self.lv3)
        #print(scell)
        
        spositions = np.zeros((np.product(scale)*self.natoms, 3))
        isat=0
        for i in range(scale[0]):
            for j in range(scale[1]):
                for k in range(scale[2]):
                    #shift ALL atoms in the unit cell
                    for iat in range(self.natoms):
                        # Be careful:
                        # e.g. for 2 atom unit cell: atm1 and atm2 in supercell 
                        #      after 2x2x1 scaling will be
                        
                        # 0 0 0 cell of supercell
                        # atom1  generated from atm1 --> same as atm1 
                        # atom2  generated from atm2 --> same as atm2

                        # 0 1 0 cell of supercell
                        # atom3   generated from atm1 
                        # atom4   generated from atm2 

                        # 1 0 0 cell of supercell
                        # atom5   generated from atm1 
                        # atom6   generated from atm2 

                        # 1 1 0 cell of supercell
                        # atom7   generated from atm1 
                        # atom8   generated from atm2
                        spositions[isat,:] = self.positions[iat,:]+i*self.lv1+j*self.lv2+k*self.lv3
                        #print(self.label[iat])
                        isat += 1
        self.spositions = spositions
        self.scell = scell
        
        
        #regroup atomic positions and labels
        # - atoms of same type are grouped together
        slabel = self.label*np.product(scale)
        satom_label, satm_id = self._get_grouped_list(slabel)

        new_spositions = np.zeros_like(spositions)
        for i in range(len(spositions)):
            new_spositions[i,:] = spositions[satm_id[i],:]
            
        #--------------
        self.new_spositions = new_spositions
        self.satom_label = satom_label
        self.scale = scale
        #---------------
        
        #ungrouped supercell atomic positions and labels: spositions and slabel
        #return spositions, scell, slabel
        #grouped supercell atomic positions and labels: new_spositions and satom_label
        return new_spositions, scell, satom_label

    
    
    def _get_unique_list(self, inp):
            '''
            get unique elements of list

            input: string: atomic labels 
            -----
            
            returns: string: uqique atomic labels for each type of atoms 
            -------

            '''
            ulist = []
            for x in inp:
                if x not in ulist:
                    ulist.append(x)
            return ulist

    def _get_grouped_list(self, llist):
        '''
        get same string elements of list grouped together
        '''

        def _get_unique_list(inp):
            '''
            get unique elements of list

            return:
                    string for each type of atoms 
            '''
            ulist = []
            for x in inp:
                if x not in ulist:
                    ulist.append(x)
            return ulist

        res = self._get_unique_list(llist)
        grouped_list = []
        idx = []

        for i in range(len(res)):
            for j in range(len(llist)):
                if res[i] == llist[j]:
                    idx.append(j)
        for i in idx:
            grouped_list.append(llist[i])

        return(grouped_list, idx)


[docs]    def get_neach_type(self, inp):
        '''
        get number of list entries of each type


        input:
        -----
             list of strings of atomic labels
        
        returns:
        -------
            get integer number of list entries (atomic symbols) of each type
        
        '''    
        glist = self._get_grouped_list(inp)[0]
        ulist = self._get_unique_list(inp)
        #print(ulist)

        neach = []
        for i in ulist:
            k = 0
            for j in glist:
                if i == j:
                    k += 1
            neach.append(k)
        #print (neach)
        return(neach)

        '''
        given input matrix (may be cell matrix or position matrix in c order) 
        and rotation matrix (optional; default is unit matrix), output respective new matrix
        
        
        parameters
        ----------
            inp_mat: array; rank 3 or rank N (no. of atoms)
            
            lcell: logical, indicates inp_mat is for cell 
            
            lposition: logical, indicates that inp_mat is for atomic points/positions
            
            translation; array/vecor; rank 1, optional (default is zero vector = (0 0 0))
            
            rotation_matrix: array of rank3, optional (default is unit matrix)
            
        returns
        -------
            inp_mat like
        
        
        Note: following convention like spglib: https://spglib.github.io/spglib/definition.html
            
            
            ====> 
            Basis vectors (a,b,c)  or (a1, a2, a3) 
            
            In spglib, basis vectors are represented by three column vectors (in Cartesian coordinates.
            ) :

            a=⎛⎝⎜ax ay az⎞⎠⎟,         b=⎛⎝⎜bx by bz⎞⎠⎟,            c=⎛⎝⎜cx cy cz⎞⎠⎟,
            
            ====>
            atomic point x are represented as three fractional values relative to basis 
            vectors as follows,
            
            x=⎛⎝⎜x1x2x3⎞⎠⎟
            
            
            ====>
            The transformation matrix P changes choice of basis vectors as follows
             (a b c) = (as bs cs) P
             where (abc) and (as bs cs) are the basis vectors of an arbitrary 
             system and of a starndardized system, respectively
             
             The origin shift p gives the vector from the origin of the standardized 
             system Os to the origin of the arbitrary system O
             p = O − Os
        
        
            A change of basis is described by the combination of the transformation
            matrix and the origin shift denoted by (P,p) where first the transformation 
            matrix is applied and then origin shift. The points in the standardized system
            xs and arbitrary system x are related by 
            
            xs = P x + p,
            
            or equivalently, 
            
            x = P^-1 xs − P^-1 p
        
        '''
        import numpy as np
        
        if lcell:
            print('applying rotation on unit cell')
        
            # inp_mat (3x3) and rotation_matrix are input in C format; => transpose ?? verify it
            return (np.matmul(inp_mat.T, rotation_matrix.T) )  #.T
