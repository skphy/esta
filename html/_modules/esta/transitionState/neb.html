        """get an estimate of the tangent at each image along the elastic band
        
        .. note::
        
           Two ways to estimation:

           1) Graeme Henkelman, et al. OURNAL OF CHEMICAL PHYSICS VOLUME 113, 
           NUMBER 22 8 DECEMBER 2000  

           2) Graeme Henkelman and H Jonsson , JOURNAL OF CHEMICAL PHYSICS VOLUME 
           113, NUMBER 22 8 DECEMBER 2000; 
           note: improved estimate needs energy of each image present in position variable
            
            
            
        parameters
        ----------
        energy: array
            array of energies (DFT or some other energy calculator) of length 
            equal to the number of images in the elastic band 
            (also need position and force ---> from self.position, self.force)
            
        returns
        -------
        tangent: array               
            array of rank 3 with shape (n_images, natoms, 3); tangent array at each inbetween 
            images to get perpendicular and parallel component of true/dft forces and spring forces
            Note: first and last tangents are kept zero and never used!!
        """        
        
        import numpy as np
        import sys
        
        energy = self.energy
        
        #NOTE: take the shape of tangent same as that of position or force
        # # first and last tangents are kept zero and never used!!
        #ang = np.zeros(( np.size(self.position, axis=0), np.shape(self.position)[1], np.shape(self.position)[2] ))
        tang = np.zeros_like(self.position)
        print('shape of the tangent: {}'.format(np.shape(tang)))
        
        #if energis is None:
        if energy is None:
            print('*** USING rough ESTIMATE OF THE TANGENT ***')
            
            # tangent at images 1 ... N-1 with corresponding position
            #tang = np.zeros(( np.size(self.position, axis=0)-2, np.shape(self.position)[1], np.shape(self.position)[2] ))
            #print('tangent shape: {}'.format(np.shape(tang)))

            
            #NOTE: take the shape of tangent same as that of position or force
            #tang = np.zeros(( np.size(self.position, axis=0), np.shape(self.position)[1], np.shape(self.position)[2] ))
            print('all tangents shape: {}'.format(np.shape(tang)))
            print('Excluding end tangents; tangent shape: {}'.format(np.shape(tang[1:-1])))
            
            
            for i in range(1, np.shape(tang)[0]-1):  # first and last tangents are kept zero and never used!!
                # i-th tangent for i-1, i, and i+1 positions
                print('tangent estimate for image: {}'.format(i))
                RiRiminus1 = self.position[i,  :,:] - self.position[i-1,:,:]  
                Riplus1Ri  = self.position[i+1,:,:] - self.position[i,  :,:]  
                tang[i,:,:] = RiRiminus1 / np.linalg.norm(RiRiminus1) + Riplus1Ri / np.linalg.norm(Riplus1Ri)
                tang[i] = tang[i]/np.linalg.norm(tang[i])
                
                
            lprint = False
            if lprint: 
                for i in range(1, np.shape(tang)[0]-1):
                    print("tangent {}: {}".format(i, tang[i]))
                    print('norm of unit tangent {}: {}'.format(i, np.linalg.norm( tang[i] )))
            
            
        else: 
            print('*** USING IMPROVED ESTIMATE OF THE TANGENT ***')
        
            # pos and energies of images 1 ... N-1
            
            
            #NOTE: take the shape of tangent same as that of position or force, and ignoring the end point
            #      tangents
            #--
            # There are four options for determining the local tangent with improved guess
            #
            #case 1; Ei+1 > Ei > Ei-1
            #case 2; Ei+1 < Ei < Ei-1
            #case 3; Ei+1 > Ei-1   }
            #case 4; Ei+1 < Ei-1   } for 3 and 4 case: Ei is either high or lower than neighbouring images
            
            #NOTE NOTE NOTE-
            # carefull about the copy command; for safety use np.zeros ...NOTE
            tang_plus = copy.copy(tang)  # energy: i+1 > i > i-1
            tang_minus = copy.copy(tang) # energy: i+1 < i < i-1
            #

            delta_Emax = np.zeros_like(energy) # first and last values Must not be used!!

            for i in range(1, np.shape(tang)[0]-1):  # first and last tangents are kept zero and never used!!
                # i-th tangent for i-1, i, and i+1 positions or images
                print('tangent estimate for image: {}'.format(i))

                tang_plus[i,:,:] =  (self.position[i+1,:,:]  - self.position[i,  :,:] )                
                tang_minus[i,:,:] = (self.position[i,  :,:]  - self.position[i-1,:,:] )

#                     delta_Emax =

                    en_cond1 = (energy[i] < energy[i+1] and energy[i] < energy[i-1])
                    en_cond2 = (energy[i] > energy[i+1] and energy[i] > energy[i-1])

                        delta_iplus1i = np.abs(energy[i+1] - energy[i])

                            #tang ...
                            print('case4 - image : : {}'.format(i))
                            tang[i, : ,:] = tang_plus[i] * delta_Emin[i] + tang_minus[i] * delta_Emax[i] 
                            tang[i] = tang[i]/np.linalg.norm(tang[i])
                        else: 
                            print('something serious wrong in the energy conditions!!!')

        """get the spring force parallel to the tangent, given the self.atomic positions and 
        spring_constant (optional)
        """
        print('//INSIDE get_spring_force_parallel//')
        
        if spring_constant is None:
            spring_constant = 5.0 # units eV/A^2; if energy in eV and atomic position in Ang unit
        
        # spring force on end points is kept zero!!
        spring_force = np.zeros_like(self.position)
        npos_set = np.shape(self.position)[0]
        
        
        tangent =  self.get_tangent()
        
        lprint1 = False
        lprint = False
        for i in range(npos_set):
            if lprint1:
                print('pos set: {}'.format(i))
                print('position: {}'.format(self.position[i,0:3,:]))
                print('spring force: {}'.format(spring_force[i,0:3,:]))
            
            if i >= 1 and i <= npos_set-2:
                print('***** i: {} '.format(i))
                
                #R{i+1} - Ri 
                Rip1Ri = (self.position[i+1] - self.position[i] )
                Rip1Ri = np.abs( np.linalg.norm(Rip1Ri) ) 
                # Ri - R{i-1}
                RiRim1 = (self.position[i] - self.position[i-1] )
                RiRim1 = np.abs ( np.linalg.norm( RiRim1))
                
#                 spring_force[i,:,:] = spring_constant * ( Rip1Ri - RiRim1 ) * self.tang[i,:,:]
                spring_force[i,:,:] = spring_constant * ( Rip1Ri - RiRim1 ) * tangent[i,:,:]
                
            if lprint:
                print('! spring force: {}'.format(spring_force[i,0:3,:]))
        
        return spring_force



[docs]    def get_force_perpen(self):
        """get the perpendicular force acting on each image excluding end points;
        
        .. note::
           force_perpendicular = force_dft  - dot_product (force_dft , tangent_vector) tangent_vector
            
           force_dft is force obtained from dft or some model calculation!
        """
        print('//INSIDE get_force_perpen//')

        force = self.force
        tangent = self.get_tangent()
        
        
        force_perpen = force - np.dot(np.ravel(force), np.ravel(tangent)) * tangent
        
        #Note: check the force on the end point images .. in case of error ..use loop on images except 
        #       end point images!!
            
        return force_perpen

    
    @property
    def get_nudged_force(self):
        print('//INSIDE get_nudged_force//')
        return self.get_spring_force_parallel() + self.get_force_perpen()
    
    @property
    def get_force_ci(self):
        """get force on the climbing image

        .. note::
           force_ci = force_dft  - 2*  dot_product (force_dft_vector , tangent_vector) tangent_vector
        """
        
        print("//INSIDE get_force_ci_image//") 
        lprint = True
        if lprint:
            print('force_ci should be called after NEB iterations - ')
            print('   when forces on images are below some theshold')
            
        force = self.force
        tangent = self.get_tangent()
        
        force_ci = force - 2* np.dot(np.ravel(force), np.ravel(tangent)) * tangent            
        return force_ci
