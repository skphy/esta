    """
    v(x,y) = Ax cos(2pix) + Ay sin(2piy)
    """
    import sys
    import numpy as np
    
    Ax = 1.
    Ay = 1.
    n = 100

    x = np.linspace(start=0, stop=4, num = n, endpoint=True)
    y = np.linspace(start=-2, stop=2, num = n, endpoint=True)

    print("inside get_potential_energy: ")
    print(len(x), len(y))
    vxy = np.zeros( (len(x), len(y) ))
    
#     xx = list(x)
#     for i in xx:
#         print(i)
#     sys.exit()
    
    
    
    
    for i, ix in enumerate(list(x)):
        for j, iy in enumerate(list(y)):
            #vxy[i,j] = Ax* np.cos(2*np.pi * ix) + Ay* np.sin(2*np.pi * iy)  
            vxy[i,j] = 10* np.cos(2*np.pi * ix) + Ay* np.sin(2*np.pi * iy)  
            #vxy[i,j] = (ix* np.cos(2*np.pi * ix)/ix) + (iy* np.sin(2*np.pi * iy)/ix)  
            
            #print(ix, iy, vxy)
            #print(ix, iy, vxy[ix,iy])
    print ('pot energy - argmin: {}'.format(np.argmin(vxy), axis=0))
    print ('pot energy - argmax: {}'.format(np.argmax(vxy), axis=0))
#     print(np.shape(vxy))
#     print(vxy)
    
    xx, yy = np.meshgrid(x, y)
    vxxyy = Ax* np.cos(2*np.pi * xx) + Ay* np.sin(2*np.pi * yy)  
    
    return (x, y, vxy), (xx, yy, vxxyy) #ix,iy
